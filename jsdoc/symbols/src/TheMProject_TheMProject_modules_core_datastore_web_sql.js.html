115     /**
116      * Saves a model in the database. Constructs the sql query from the model record. Prepares an INSERT or UPDATE depending on the state
117      * of the model. If M.STATE_NEW then prepares an INSERT, if M.STATE_VALID then prepares an UPDATE. The operation itself
118      * is done by {@link M.DataProviderWebSql#performOp} that is called
119      *
120      * @param {Object} obj The param obj, includes:
121      * * onSuccess callback
122      * * onError callback
123      * * the model
124      */
125      save: function(obj) {
126         //console.log('save() called.');
127 
128         this.onSuccess = obj.onSuccess;
129         this.onError = obj.onError;
130 
131         /**
132          * if not already done, initialize db/table first
133          */
134         if(!this.isInitialized) {
135             this.internalCallback = this.save;
136             this.init(obj);
137             return;
138         }
139 
140         var pre_suffix = '';
141         var sif
142 
143         if(obj.model.state === M.STATE_NEW) { // perform an INSERT
144 
145             var sql = 'INSERT INTO ' + obj.model.name + ' (';
146             for(var prop in obj.model.record) {
147                 sql += prop + ', ';
148             }
149 
150             /* now name m_id column */
151             sql += M.META_M_ID + ') ';
152 
153             /* VALUES(12, 'Test', ... ) */
154             sql += 'VALUES (';
155 
156             for(var prop2 in obj.model.record) {
157                 //console.log(obj.model.record[prop2]);
158                 var propDataType = obj.model.__meta[prop2].dataType;
159                 /* if property is string or text write value in quotes */
160                 pre_suffix = propDataType === 'String' || propDataType === 'Text' || propDataType === 'Date' ? '"' : '';
161                 /* if property is date object, convert to string by calling toJSON */
162                 recordPropValue = (obj.model.record[prop2].type === 'M.Date') ? obj.model.record[prop2].toJSON() : obj.model.record[prop2];
163                 sql += pre_suffix + recordPropValue + pre_suffix + ', ';
164             }
165 
166             sql += obj.model.m_id + ')';
167 
168             this.performOp(sql, obj, 'INSERT');
169 
170         } else { // perform an UPDATE with id of model
171 
172             var sql = 'UPDATE ' + obj.model.name + ' SET ';
173 
174             var nrOfUpdates = 0;
175 
176             for(var p in obj.model.record) {
177 
178                 if(p === 'ID' || !obj.model.__meta[p].isUpdated) { /* if property has not been updated, then exclude from update call */
179                     continue;
180                 }
181                 nrOfUpdates = nrOfUpdates + 1;
182 
183                 pre_suffix = obj.model.__meta[p].dataType === 'String' || obj.model.__meta[p].dataType === 'Text' || obj.model.__meta[p].dataType === 'Date' ? '"' : '';
184 
185                 /* if property is date object, convert to string by calling toJSON */
186                 recordPropValue = obj.model.__meta[p].dataType === 'Date' ? obj.model.record[p].toJSON() : obj.model.record[p];
187 
188                 sql += p + '=' + pre_suffix + recordPropValue + pre_suffix + ', ';
189             }
190             sql = sql.substring(0, sql.lastIndexOf(','));
191             sql += ' WHERE ' + 'ID=' + obj.model.record.ID + ';';
192 
193             /* if no properties updated, do nothing, just return by calling onSuccess callback */
194             if(nrOfUpdates === 0) {
195                 if (obj.onSuccess && obj.onSuccess.target && obj.onSuccess.action) {
196                     obj.onSuccess = this.bindToCaller(obj.onSuccess.target, obj.onSuccess.target[obj.onSuccess.action], null, obj, this);
197                     obj.onSuccess();
198                 }else if(obj.onSuccess && typeof(obj.onSuccess) === 'function') {
199                     obj.onSuccess(result);
200                 }
201             } else {
202                 this.performOp(sql, obj, 'UPDATE');
203             }
204         }

207     /**
208      * Performs operation on WebSQL storage: INSERT, UPDATE or DELETE. Is used by {@link M.WebSqlProvider#save} and {@link M.WebSqlProvider#del}.
209      * Calls is made asynchronously, means that result is just available in callback.
210      * @param {String} sql The query.
211      * @param {Object} obj The param object. Contains e.g. callbacks (onError & onSuccess)
212      * @param {String} opType The String identifying the operation: 'INSERT', 'UPDATE' oder 'DELETE'
213      * @private
214      */
215     performOp: function(sql, obj, opType) {
216 
217         var that = this;
218         this.dbHandler.transaction(function(t) {
219             t.executeSql(sql, null, function() {
220                 if(opType === 'INSERT') { /* after INSERT operation set the assigned DB ID to the model records id */
221                     that.queryDbForId(obj.model);
222                 }
223             }, function() { // error callback for SQLStatementTransaction
224                 M.Logger.log('Incorrect statement: ' + sql, M.ERR);
225             });
226         },
227         function(sqlError) { // errorCallback
228 
229             var err = this.buildErrorObject(sqlError);
230             this.handleErrorCallback(obj, err);
231         },
232 
233         function() {    // voidCallback (success)
234              /* delete  the model from the model record list */
235             if(opType === 'DELETE') {
236                 obj.model.recordManager.remove(obj.model.m_id);
237             }
238             /* bind success callback */
239             if (obj.onSuccess && obj.onSuccess.target && obj.onSuccess.action) {
240                 obj.onSuccess = that.bindToCaller(obj.onSuccess.target, obj.onSuccess.target[obj.onSuccess.action], null, obj, this);
241                 obj.onSuccess();
242             }else if(obj.onSuccess && typeof(obj.onSuccess) === 'function') {
243                 obj.onSuccess(null, this);
244             }
245         });
246     },

271     /**
272      * Finds model records in the database. If a constraint is given, result is filtered.
273      * @param {Object} obj The param object. Includes:
274      * * model: the model blueprint
275      * * onSuccess:
276      * * onError:
277      * * columns: Array of strings naming the properties to be selected: ['name', 'age'] => SELECT name, age FROM...
278      * * constraint: Object containing itself two properties:
279      *      * statement: a string with the statement, e.g. 'WHERE ID = ?'
280      *      * parameters: array of strings with the parameters, length array must be the same as the number of ? in statement
281      *          => ? are substituted with the parameters
282      * * order: String with the ORDER expression: e.g. 'ORDER BY price ASC'
283      * * limit: Number defining the number of max. result items
284      */
285     find: function(obj) {
286 
287         this.onSuccess = obj.onSuccess;
288         this.onError = obj.onError;
289 
290         if(!this.isInitialized) {
291             this.internalCallback = this.find;
292             this.init(obj, this.bindToCaller(this, this.find));
293             return;
294         }
295 
296         var sql = 'SELECT ';
297 
298         if(obj.columns) {
299             /* ID column always needs to be in de result relation */
300             if(!(_.include(obj.columns, 'ID'))) {
301                 obj.columns.push('ID');
302             }
303 
304             if(obj.columns.length > 1) {
305                 sql += obj.columns.join(', ');
306             } else if(obj.columns.length == 1) {
307                 sql += obj.columns[0] + ' ';
308             }
309         } else {
310             sql += '* ';
311         }
312 
313         sql += ' FROM ' + obj.model.name + ' ';
314 
315         var stmtParameters = [];
316 
317         /* now process constraint */
318         if(obj.constraint) {
319 
320             var n = obj.constraint.statement.split("?").length - 1;
321             //console.log('n: ' + n);
322             /* if parameters are passed we assign them to stmtParameters, the array that is passed for prepared statement substitution*/
323             if(obj.constraint.parameters) {
324 
325                 if(n === obj.constraint.parameters.length) { /* length of parameters list must match number of ? in statement */
326                     sql += obj.constraint.statement;
327                     stmtParameters = obj.constraint.parameters;
328                 } else {
329                     M.Logger.log('Not enough parameters provided for statement: given: ' + obj.constraint.parameters.length + ' needed: ' + n, M.ERR);
330                     return NO;
331                 }
332            /* if no ? are in statement, we handle it as a non-prepared statement
333             * => developer needs to take care of it by himself regarding
334             * sql injection
335             */
336             } else if(n === 0) {
337                 sql += obj.constraint.statement;
338             }
339         }
340 
341         /* now attach order */
342         if(obj.order) {
343             sql += ' ORDER BY ' + obj.order
344         }
345 
346         /* now attach limit */
347         if(obj.limit) {
348             sql += ' LIMIT ' + obj.limit
349         }
350 
351         var result = [];
352         var that = this;
353         this.dbHandler.readTransaction(function(t) {
354             t.executeSql(sql, stmtParameters, function (tx, res) {
355                 var len = res.rows.length;//, i;
356                 for (var i = 0; i < len; i++) {
357                     var rec = JSON.parse(JSON.stringify(res.rows.item(i))); /* obj returned form WebSQL is non-writable, therefore needs to be converted */
358                     /* set m_id property of record to m_id got from db, then delete m_id property named after db column (M.META_M_ID) */
359                     rec['m_id'] = rec[M.META_M_ID];
360                     delete rec[M.META_M_ID];
361                     /* create model record from result with state valid */
362                     /* $.extend merges param1 object with param2 object*/
363                     var myRec = obj.model.createRecord($.extend(rec, {state: M.STATE_VALID}));
364 
365                     /* create M.Date objects for all date properties */
366                     for(var j in myRec.__meta) {
367                         /* here we can work with setter and getter because myRec already is a model record */
368                         if(myRec.__meta[j].dataType === 'Date' && typeof(myRec.get(j)) === 'string') {
369                             myRec.set(j, M.Date.create(myRec.get(j)));
370                         }
371                     }
372                     /* add to result array */
373                     result.push(myRec);
374                 }
375 
376             }, function(){M.Logger.log('Incorrect statement: ' + sql, M.ERR)}) // callbacks: SQLStatementErrorCallback
377         }, function(sqlError){ // errorCallback
378             var err = this.buildErrorObject(sqlError);
379             this.handleErrorCallback(obj, err);
380         }, function() { // voidCallback (success)
381             /* bind success callback */
382             if(obj.onSuccess && obj.onSuccess.target && obj.onSuccess.action) {
383                 /* [result] is a workaround for bindToCaller: bindToCaller uses call() instead of apply() if an array is given.
384                  * result is an array, but what call is doing with it is wrong in this case. call maps each array element to one method
385                  * parameter of the function called. Our callback only has one parameter so it would just pass the first value of our result set to the
386                  * callback. therefor we now put result into an array (so we have an array inside an array) and result as a whole is now passed as the first
387                  * value to the callback.
388                  *  */
389                 obj.onSuccess = that.bindToCaller(obj.onSuccess.target, obj.onSuccess.target[obj.onSuccess.action], [result], obj, this);
390                 obj.onSuccess();
391             }else if(obj.onSuccess && typeof(obj.onSuccess) === 'function') {
392                 obj.onSuccess(result, obj, this);
393             }
394         });
395     },

401     openDb: function() {
402         /* openDatabase(db_name, version, description, estimated_size, callback) */
403         try {
404             if (!window.openDatabase) {
405                 M.DialogView.alert({
406                     message: 'Your browser does not support WebSQL.',
407                     title: 'WebSQL Not Supported.'
408                 });
409             } else {
410                 //this.dbHandler = openDatabase(this.config.dbName, '1.0', 'Database for ' + M.Application.name, this.config.size);
411                 /* leave version empty to open database regardless of its version */
412                 if(!this.dbHandler) {
413                     this.dbHandler = openDatabase(this.config.dbName, '', 'Database for ' + M.Application.name, this.config.size);
414                 }
415 
416             }
417         } catch(e) {
418 
419             if (e == 2) {
420                 // Version number mismatch.
421                 //M.Logger.log('Invalid database version.', M.ERR);
422                 M.DialogView.alert({
423                     message: 'Database version 1.0 not supported.',
424                     title: 'Invalid database version.'
425                 });
426             } else {
427                 //M.Logger.log('Unknown error ' + e + '.', M.ERR);
428                 M.DialogView.alert({
429                     message: e,
430                     title: 'Unknown error.'
431                 });
432             }
433             return;
434         }

440     /**
441      * Creates the table corresponding to the model record.
442      * @private
443      */
444     createTable: function(obj, callback) {
445         var sql = 'CREATE TABLE IF NOT EXISTS '  + obj.model.name
446                     + ' (ID INTEGER PRIMARY KEY ASC AUTOINCREMENT UNIQUE';
447 
448         for(var r in obj.model.__meta) {
449             if(r === 'ID') {/* skip ID, it is defined manually above */
450                 continue;
451             }
452            sql += ', ' + this.buildDbAttrFromProp(obj.model, r);
453         }
454 
455         sql += ', ' + M.META_M_ID + ' INTEGER NOT NULL);';
456 
457         if(this.dbHandler) {
458             var that = this;
459             try {
460                 /* transaction has 3 parameters: the transaction callback, the error callback and the success callback */
461                 this.dbHandler.transaction(function(t) {
462                     t.executeSql(sql);
463                 }, function(sqlError){ // errorCallback
464                     var err = this.buildErrorObject(sqlError);
465                     this.handleErrorCallback(obj, err);
466                 });
467             } catch(e) {
468                 var err = this.buildErrorObject(e);
469                 this.handleErrorCallback(obj, err);
470             }
471         } else {
472             var err = this.buildErrorObject({
473                 code: 8,
474                 message: 'dbHandler does not exist.'
475             });
476             this.handleErrorCallback(obj, err);
477             M.Logger.log('dbHandler does not exist.', M.ERR);
478         }
479     },

481     /**
482      *
483      * @param {obj} obj
484      * @param {Array} records
485      * @param {Number} transactionSize
486      */
487     bulkImport: function(obj) {
488 
489         if(!this.isInitialized) {
490             this.internalCallback = this.bulkImport;
491             this.init(obj);
492             return;
493         }
494 
495         /* reset flag */
496         this.bulkTransactionFailed = NO;
497 
498         var records = obj.records;
499         /* if no records were passed execute error callback and pass it an error object */
500         if(!records || !_.isArray(records)) {
501             var err = this.buildErrorObject({
502                 code: 10,
503                 message: 'No Records array passed.'
504             });
505             this.handleErrorCallback(obj, err);
506         }
507         /* if no transactionSize was passed, make one transaction for all */
508         var transactionSize = !obj.transactionSize || typeof(transactionSize) !== 'number' ? records.length : obj.transactionSize;
509 
510         var transactions = [];
511 
512         var sqlTemplate = ' INSERT OR REPLACE INTO ' + obj.model.name + ' (';
513         /* append columns */
514         for(var prop in obj.model.record) {
515             sqlTemplate += prop + ', ';
516         }
517 
518         /* append m_id column */
519         sqlTemplate += M.META_M_ID + ') ';
520         /* now append values */
521         sqlTemplate += 'VALUES (';
522 
523         var sql = ''; // the actual sql insert string with values
524         var values = '';
525         var curRec = null;
526         for (var i = 1; i <= records.length; i++) {
527             curRec = records[i - 1];
528             for(var prop2 in curRec.record) {
529                 var propDataType = curRec.__meta[prop2].dataType;
530                 /* if property is string or text write value in quotes */
531                 var pre_suffix = propDataType === 'String' || propDataType === 'Text' || propDataType === 'Date' ? '"' : '';
532                 /* if property is date object, convert to string by calling toJSON */
533                 var recordPropValue = (curRec.record[prop2].type === 'M.Date') ? curRec.record[prop2].toJSON() : curRec.record[prop2];
534                 values += pre_suffix + recordPropValue + pre_suffix + ', ';
535             }
536             sql += sqlTemplate + values + curRec.m_id + ');';
537             values = '';
538 
539             /* save sql string as one transaction pack */
540             if (i % transactionSize === 0 || i === records.length) {
541                 transactions.push(sql);
542                 sql = '';
543             }
544         }
545 
546         for(var i = 0; i < transactions.length;i++) {
547             this.makeBulkTransaction(obj, transactions[i], i, transactions.length);
548         }
549 
550     },

552     makeBulkTransaction: function(obj, transaction, transactionNr, numberOfTransactions) {
553         var that = this;
554         var sqlInsertSet = transaction.split(';');
555         if(!this.bulkTransactionFailed) {
556             if(this.dbHandler) {
557                 /* transaction has 3 parameters: the transaction callback, the error callback and the success callback */
558                 this.dbHandler.transaction(function(t) {
559                     for(var j = 0; j < sqlInsertSet.length; j++) {
560                         if(sqlInsertSet[j] !== '') {
561                             console.log(sqlInsertSet[j]);
562                             t.executeSql(sqlInsertSet[j]);
563                         }
564                     }
565                 }, function(sqlError) { // errorCallback
566                     that.bulkTransactionFailed = YES;
567                     var err = that.buildErrorObject(sqlError);
568                     this.handleErrorCallback(obj, sqlError);
569                 },
570                 function() {
571                     that.handleBulkSuccessCallback(obj, (transactionNr + 1), numberOfTransactions);
572                 });
573             } else {
574                 M.Logger.log('dbHandler does not exist.', M.ERR);
575             }
576         } else {
577             return NO;
578         }
579     },

582     handleBulkSuccessCallback: function(obj, iteration, total) {
583         if(iteration === total) {
584              if(obj.onSuccess && obj.onSuccess.target && obj.onSuccess.action) {
585                 obj.onSuccess = that.bindToCaller(obj.onSuccess.target, obj.onSuccess.target[obj.onSuccess.action], null, obj, this);
586                 obj.onSuccess();
587             }else if(obj.onSuccess && typeof(obj.onSuccess) === 'function') {
588                 obj.onSuccess(null, obj, this);
589             }
590         }
591     },
592 
593     handleErrorCallback: function(obj, err) {
594         if (obj.onError && obj.onError.target && obj.onError.action) {
595                 obj.onError = that.bindToCaller(obj.onError.target, obj.onError.target[obj.onError.action], err, obj, this);
596                 obj.onError();
597             } else if (typeof(obj.onError) === 'function') {
598                 obj.onError(err, obj, this);
599             } else {
600                 M.Logger.log('Target and action in onError not defined.', M.ERR);
601             }
602     },

670     /**
671      * @private
672      * Builds a M.Error object on basis of SQLError sqlErr. Maps error codes to M.Error error codes.
673      * @param {Object} err err object returned by WebSQL or manually created error obj
674      */

 11 /**
 12  * @class
 13  *
 14  * M.PageView is the prototype of any page. It is the seconds 'highest' view, right after
 15  * M.Application. A page is the container view for all other views.
 16  *
 17  * @extends M.View
 18  */
 19 M.PageView = M.View.extend(
 20 /** @scope M.PageView.prototype */ {
 21 
 22     /**
 23      * The type of this object.
 24      *
 25      * @type String
 26      */
 27     type: 'M.PageView',
 28 
 29     /**
 30      * States whether a page is loaded the first time or not. It is automatically set to NO
 31      * once the page was first loaded.
 32      *
 33      * @type Boolean
 34      */
 35     isFirstLoad: YES,
 36 
 37     /**
 38      * This property can be used to set the page's beforeLoad action.
 39      *
 40      * @type Object
 41      */
 42     beforeLoad: null,
 43 
 44     /**
 45      * This property can be used to set the page's onLoad action.
 46      *
 47      * @type Object
 48      */
 49     onLoad: null,
 50 
 51     /**
 52      * This property can be used to set the page's beforeHide action.
 53      *
 54      * @type Object
 55      */
 56     beforeHide: null,
 57 
 58     /**
 59      * This property can be used to set the page's onHide action.
 60      *
 61      * @type Object
 62      */
 63     onHide: null,
 64 
 65     /**
 66      * This property can be used to set the page's onOrientationChange action.
 67      *
 68      * @type Object
 69      */
 70     onOrientationChange: null,
 71 
 72     /**
 73      * Indicates whether the page has a tab bar or not.
 74      *
 75      * @type Boolean
 76      */
 77     hasTabBarView: NO,
 78 
 79     /**
 80      * The page's tab bar.
 81      *
 82      * @type M.TabBarView
 83      */
 84     tabBarView: null,
 85 
 86     /**
 87      * This property specifies the recommended events for this type of view.
 88      *
 89      * @type Array
 90      */
 91     recommendedEvents: ['pagebeforeshow', 'pageshow', 'pagebeforehide', 'pagehide'],
 92 
 93     /**
 94      * This property is used to specify a view's internal events and their corresponding actions. If
 95      * there are external handlers specified for the same event, the internal handler is called first.
 96      *
 97      * @type Object
 98      */
 99     internalEvents: null,
100 
101     /**
102      * An associative array containing all list views used in this page. The key for a list view is
103      * its id. We do this to have direct access to a list view, so we can reset its selected item
104      * once the page was hidden.
105      *
106      * @type Object
107      */
108     listList: null,
109 
110     /**
111      * Renders in three steps:
112      * 1. Rendering Opening div tag with corresponding data-role
113      * 2. Triggering render process of child views
114      * 3. Rendering closing tag
115      *
116      * @private
117      * @returns {String} The page view's html representation.
118      */
119     render: function() {
120         /* store the currently rendered page as a reference for use in child views */
121         M.ViewManager.currentlyRenderedPage = this;
122         
123         this.html += '<div id="' + this.id + '" data-role="page"' + this.style() + '>';
124 
125         this.renderChildViews();
126 
127         this.html += '</div>';
128 
129         this.writeToDOM();
130         this.theme();
131         this.registerEvents();
132     },
133 
134     /**
135      * This method is responsible for registering events for view elements and its child views. It
136      * basically passes the view's event-property to M.EventDispatcher to bind the appropriate
137      * events.
138      *
139      * It extend M.View's registerEvents method with some special stuff for page views and its
140      * internal events.
141      */
142     registerEvents: function() {
143         this.internalEvents = {
144             pagebeforeshow: {
145                 target: this,
146                 action: 'pageWillLoad'
147             },
148             pageshow: {
149                 target: this,
150                 action: 'pageDidLoad'
151             },
152             pagebeforehide: {
153                 target: this,
154                 action: 'pageWillHide'
155             },
156             pagehide: {
157                 target: this,
158                 action: 'pageDidHide'
159             }
160         }
161         this.bindToCaller(this, M.View.registerEvents)();
162     },
163 
164     /**
165      * This method writes the view's html string into the DOM. M.Page is the only view that does
166      * that. All other views just deliver their html representation to a page view.
167      */
168     writeToDOM: function() {
169         document.write(this.html);
170     },
171 
172     /**
173      * This method is called right before the page is loaded. If a beforeLoad-action is defined
174      * for the page, it is now called.
175      *
176      * @param {String} id The DOM id of the event target.
177      * @param {Object} event The DOM event.
178      * @param {Object} nextEvent The next event (external event), if specified.
179      */
180     pageWillLoad: function(id, event, nextEvent) {
181         /* initialize the tabbar */
182         if(M.Application.isFirstLoad) {
183             M.Application.isFirstLoad = NO;
184             var currentPage = M.ViewManager.getCurrentPage();
185             if(currentPage && currentPage.hasTabBarView) {
186                 var tabBarView = currentPage.tabBarView;
187 
188                 if(tabBarView.childViews) {
189                     var childViews = tabBarView.getChildViewsAsArray();
190                     for(var i in childViews) {
191                         if(M.ViewManager.getPage(tabBarView[childViews[i]].page).id === currentPage.id) {
192                             tabBarView.setActiveTab(tabBarView[childViews[i]]);
193                         }
194                     }
195                 }
196             }
197         }
198 
199         /* initialize the loader for later use (if not already done) */
200         if(M.LoaderView) {
201             M.LoaderView.initialize();
202         }
203 
204         /* reset the page's title */
205         document.title = M.Application.applicationTitle;
206 
207         /* delegate event to external handler, if specified */
208         if(nextEvent) {
209             M.EventDispatcher.callHandler(nextEvent, event, NO, [this.isFirstLoad]);
210         }
211     },
212 
213     /**
214      * This method is called right after the page was loaded. If a onLoad-action is defined
215      * for the page, it is now called.
216      *
217      * @param {String} id The DOM id of the event target.
218      * @param {Object} event The DOM event.
219      * @param {Object} nextEvent The next event (external event), if specified.
220      */
221     pageDidLoad: function(id, event, nextEvent) {
222         /* delegate event to external handler, if specified */
223         if(nextEvent) {
224             M.EventDispatcher.callHandler(nextEvent, event, NO, [this.isFirstLoad]);
225         }
226 
227         /* call jqm to fix header/footer */
228         $.fixedToolbars.show();
229 
230         this.isFirstLoad = NO;
231     },
232 
233     /**
234      * This method is called right before the page is hidden. If a beforeHide-action is defined
235      * for the page, it is now called.
236      *
237      * @param {String} id The DOM id of the event target.
238      * @param {Object} event The DOM event.
239      * @param {Object} nextEvent The next event (external event), if specified.
240      */
241     pageWillHide: function(id, event, nextEvent) {
242         /* delegate event to external handler, if specified */
243         if(nextEvent) {
244             M.EventDispatcher.callHandler(nextEvent, event, NO, [this.isFirstLoad]);
245         }
246     },
247 
248     /**
249      * This method is called right after the page was hidden. If a onHide-action is defined
250      * for the page, it is now called.
251      *
252      * @param {String} id The DOM id of the event target.
253      * @param {Object} event The DOM event.
254      * @param {Object} nextEvent The next event (external event), if specified.
255      */
256     pageDidHide: function(id, event, nextEvent) {
257         /* if there is a list on the page, reset it: deactivate possible active list items */
258         if(this.listList) {
259             _.each(this.listList, function(list) {
260                 list.resetActiveListItem();
261             });
262         }
263 
264         /* delegate event to external handler, if specified */
265         if(nextEvent) {
266             M.EventDispatcher.callHandler(nextEvent, event, NO, [this.isFirstLoad]);
267         }
268     },
269 
270     /**
271      * Triggers the rendering engine, jQuery mobile, to style the page and call the theme() of
272      * its child views.
273      *
274      * @private
275      */
276     theme: function() {
277         $('#' + this.id).page();
278         this.themeChildViews();
279     },
280 
281     /**
282      * Applies some style-attributes to the page.
283      *
284      * @private
285      * @returns {String} The page's styling as html representation.
286      */
287     style: function() {
288         var html = '';
289         if(this.cssClass) {
290             if(!html) {
291                 html += ' class="';
292             }
293             html += this.cssClass;
294         }
295         if(html) {
296             html += '"';
297         }
298         return html;
299     }
300     
301 });

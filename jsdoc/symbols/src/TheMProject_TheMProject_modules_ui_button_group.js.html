 11 /**
 12  * A constant value for horizontal alignment.
 13  *
 14  * @type String
 15  */
 16 M.HORIZONTAL = 'horizontal';
 17 
 18 /**
 19  * A constant value for vertical alignment.
 20  *
 21  * @type String
 22  */
 23 M.VERTICAL = 'vertical';

 26 /**
 27  * @class
 28  *
 29  * A button group is a vertically or / and horizontally aligned group of buttons. There
 30  * are basically three different types of a button group:
 31  *
 32  * - horizontally aligned buttons
 33  *     1 - 2 - 3
 34  *
 35  * - vertically aligned buttons
 36  *     1
 37  *     |
 38  *     2
 39  *     |
 40  *     3
 41  *
 42  * - horizontally and vertically aligned buttons
 43  *     1 - 2
 44  *     |   |
 45  *     3 - 4
 46  * 
 47  * @extends M.View
 48  */
 49 M.ButtonGroupView = M.View.extend(
 50 /** @scope M.ButtonGroupView.prototype */ {
 51 
 52     /**
 53      * The type of this object.
 54      *
 55      * @type String
 56      */
 57     type: 'M.ButtonGroupView',
 58 
 59     /**
 60      * This property determines whether to render the button group horizontally
 61      * or vertically. Default: horizontal.
 62      *
 63      * Possible values are:
 64      * - M.HORIZONTAL: horizontal
 65      * - M.VERTICAL: vertical
 66      *
 67      * @type String
 68      */
 69     direction: M.HORIZONTAL,
 70 
 71     /**
 72      * Determines whether to display the button group view 'inset' or at full width.
 73      *
 74      * @type Boolean
 75      */
 76     isInset: YES,
 77 
 78     /**
 79      * Determines whether to display the button group compact, i.e. without top/bottom
 80      * margin. This property only is relevant in combination with multiple lines of
 81      * buttons (c.p.: buttonsPerLine property).
 82      *
 83      * @type Boolean
 84      */
 85     isCompact: YES,
 86 
 87     /**
 88      * This property, if set, defines how many buttons are rendered per line. If there
 89      * are more buttons defined that fitting into one line, the following buttons are
 90      * rendered into a new line. Make sure, the number of your buttons is divisible by
 91      * the number of buttons per line, since only full lines are displayed. So if you
 92      * for example specify 5 buttons and 2 buttons per line, the fifth button won't be
 93      * visible.
 94      *
 95      * If e.g. 4 buttons are specified and this property is set to 2, the rendering will
 96      * be as follows:
 97      *
 98      *     1 -- 2
 99      *     3 -- 4
100      *
101      * @type Number
102      */
103     buttonsPerLine: null,
104 
105     /**
106      * This property is used to internally store the number of lines that are necessary
107      * to render all buttons according to the buttonsPerLine property.
108      *
109      * @private
110      * @type Number
111      */
112     numberOfLines: null,
113 
114     /**
115      * This property refers to the currently rendered line, if there is more than one.
116      *
117      * @private
118      * @type Number
119      */
120     currentLine: null,
121 
122     /**
123      * This property contains an array of html ids referring to the several lines of grouped
124      * buttons, if there is more than one at all.
125      *
126      * @private
127      * @type Array
128      */
129     lines: null,
130 
131     /**
132      * This property contains a reference to the currently selected button.
133      *
134      * @private
135      * @type Object
136      */
137     activeButton: null,
138 
139     /**
140      * This property determines whether the buttons of this button group are selectable or not. If
141      * set to YES, a click on one of the buttons will set this button as the currently active button
142      * and automatically change its styling to visualize its selection.
143      *
144      * @type Boolean
145      */
146     isSelectable: YES,
147 
148     /**
149      * This property specifies the recommended events for this type of view.
150      *
151      * @type Array
152      */
153     recommendedEvents: ['change'],
154 
155     /**
156      * Renders a button group as a div container and calls the renderChildViews
157      * method to render the included buttons.
158      *
159      * @private
160      * @returns {String} The button group view's html representation.
161      */
162     render: function() {
163         /* check if multiple lines are necessary before rendering */
164         if(this.childViews) {
165             var childViews = this.getChildViewsAsArray();
166             if(this.buttonsPerLine && this.buttonsPerLine < childViews.length) {
167                 var numberOfButtons = 0;
168                 for(var i in childViews) {
169                     if(this[childViews[i]] && this[childViews[i]].type === 'M.ButtonView') {
170                         numberOfButtons = numberOfButtons + 1;
171                     }
172                 }
173                 if(this.buttonsPerLine < numberOfButtons) {
174                     this.numberOfLines = M.Math.round(numberOfButtons / this.buttonsPerLine, M.FLOOR);
175                 }
176             }
177         }
178 
179         /* if there are multiple lines, render multiple horizontally aligned button groups */
180         if(this.numberOfLines) {
181             /* set the direction to horizontally, no matter what it was set to before */
182             this.direction = M.HORIZONTAL;
183 
184             /* this is a wrapper for the multiple button groups.
185                if it is not inset, assign css class 'ui-listview' for clearing the padding of the surrounding element */
186             this.html += '<div id="' + this.id + '"';
187             this.html += this.style();
188             this.html += '>';
189 
190             /* create a button group for every line */
191             this.lines = [];
192             for(var i = 0; i < this.numberOfLines; i++) {
193                 this.currentLine = i + 1;
194                 /* store current line in lines property for use in renderChildViews() */
195                 this.lines.push(this.id + '_' + i);
196 
197                 this.html += '<div data-role="controlgroup" href="#" id="' + this.id + '_' + i + '" data-type="' + this.direction + '"';
198 
199                 /* if isCompact, assign specific margin, depending on line number (first, last, other) */
200                 if(!this.isInset || this.isCompact) {
201                     if(i == 0) {
202                         this.html += this.isInset ? ' style="margin-bottom:0px"' : ' style="margin:0px"';
203                     } else if(i > 0 && i < this.numberOfLines - 1) {
204                         this.html += this.isInset ? ' style="margin:0px 0px 0px 0px"' : ' style="margin:0px"';
205                     } else if(i < this.numberOfLines) {
206                         this.html += this.isInset ? ' style="margin-top:0px"' : ' style="margin:0px"';
207                     }
208                 }
209 
210                 this.html += '>';
211 
212                 /* render the buttons for the current line */
213                 this.renderChildViews();
214 
215                 this.html += '</div>';
216             }
217             this.html += '</div>';
218         } else {
219             this.html += '<div data-role="controlgroup" href="#" id="' + this.id + '" data-type="' + this.direction + '"' + this.style() + '>';
220 
221             this.renderChildViews();
222 
223             this.html += '</div>';
224         }
225 
226         return this.html;
227     },
228 
229     /**
230      * Triggers render() on all children of type M.ButtonGroupView.
231      *
232      * @private
233      */
234     renderChildViews: function() {
235         if(this.childViews) {
236             var childViews = this.getChildViewsAsArray();
237             var currentButtonIndex = 0;
238 
239             for(var i in childViews) {
240                 if(this[childViews[i]] && this[childViews[i]].type === 'M.ButtonView') {
241                     currentButtonIndex = currentButtonIndex + 1;
242 
243                     if(!this.numberOfLines || M.Math.round(currentButtonIndex / this.buttonsPerLine, M.CEIL) === this.currentLine) {
244 
245                         var button = this[childViews[i]];
246                         /* reset buttons html, to make sure it doesn't get rendered twice if this is multi button group */
247                         button.html = '';
248 
249                         button.parentView = this;
250                         button.internalEvents = {
251                             tap: {
252                                 target: this,
253                                 action: 'setActiveButton'
254                             }
255                         }
256 
257                         /* if the buttons are horizontally aligned, compute their width depending on the number of buttons
258                            and set the right margin to '-2px' since the jQuery mobile default would cause an ugly gap to
259                            the right of the button group */
260                         if(this.direction === M.HORIZONTAL) {
261                             button.cssStyle = 'margin-right:-2px;width:' + 100 / (this.numberOfLines ? this.buttonsPerLine : childViews.length) + '%';
262                         }
263 
264                         /* set the button's _name property */
265                         this[childViews[i]]._name = childViews[i];
266 
267                         /* finally render the button and add it to the button groups html */
268                         this.html += this[childViews[i]].render();
269                     }
270                 } else {
271                     M.Logger.log('childview of button group is no button.', M.WARN);
272                 }
273             }
274         }
275     },
276 
277     /**
278      * This method themes the button group and activates one of the included buttons
279      * if its isActive property is set.
280      *
281      * @private
282      */
283     theme: function() {
284         /* if there are multiple lines of buttons, it's getting heavy */
285         if(this.numberOfLines) {
286             
287             /* iterate through all lines */
288             for(var line in this.lines) {
289                 line = parseInt(line);
290 
291                 /* style the current line */
292                 $('#' + this.lines[line]).controlgroup();
293                 var childViews = this.getChildViewsAsArray();
294                 var currentButtonIndex = 0;
295                 
296                 /* if isCompact, iterate through all buttons */
297                 if(this.isCompact) {
298                     for(var i in childViews) {
299                         i = parseInt(i);
300                         if(this[childViews[i]] && this[childViews[i]].type === 'M.ButtonView') {
301                             currentButtonIndex = currentButtonIndex + 1;
302                             var currentLine = M.Math.round(currentButtonIndex / this.buttonsPerLine, M.CEIL) - 1;
303                             var button = this[childViews[i]];
304 
305                             /* if the button belongs to the current line adjust its styling according to its position,
306                                e.g. the first button in the first row gets the css class 'ui-corner-tl' (top left). */
307                             if(line === currentLine) {
308 
309                                 /* first line */
310                                 if(line === 0 && this.numberOfLines > 1) {
311                                     /* first button */
312                                     if(currentButtonIndex === 1) {
313                                         $('#' + button.id).removeClass('ui-corner-left');
314                                         if(this.isInset) {
315                                             $('#' + button.id).addClass('ui-corner-tl');
316                                         }
317                                     /* last button */
318                                     } else if(currentButtonIndex === this.buttonsPerLine) {
319                                         $('#' + button.id).removeClass('ui-corner-right');
320                                         if(this.isInset) {
321                                             $('#' + button.id).addClass('ui-corner-tr');
322                                         }
323                                     }
324                                 /* last line */
325                                 } else if(line === this.numberOfLines - 1) {
326                                     /* first button */
327                                     if(currentButtonIndex === (currentLine * this.buttonsPerLine) + 1) {
328                                         $('#' + button.id).removeClass('ui-corner-left');
329                                         $('#' + button.id).addClass('ui-corner-bl');
330                                     /* last button */
331                                     } else if(currentButtonIndex === ((currentLine + 1) * this.buttonsPerLine)) {
332                                         $('#' + button.id).removeClass('ui-corner-right');
333                                         $('#' + button.id).addClass('ui-corner-br');
334                                     }
335                                 /* all other lines */
336                                 } else {
337                                     /* first button */
338                                     if(currentButtonIndex === (currentLine * this.buttonsPerLine) + 1) {
339                                         $('#' + button.id).removeClass('ui-corner-left');
340                                     /* last button */
341                                     } else if(currentButtonIndex === ((currentLine + 1) * this.buttonsPerLine)) {
342                                         $('#' + button.id).removeClass('ui-corner-right');
343                                     }
344                                 }
345                             }
346                         }
347                     }
348                 }
349             }
350         /* if there is only on row, simply style that button group */
351         } else {
352             $('#' + this.id).controlgroup();
353         }
354 
355         /* iterate through all buttons and activate on of them, according to the button's isActive property */
356         if(this.childViews) {
357             var childViews = this.getChildViewsAsArray();
358             for(var i in childViews) {
359                 if(this[childViews[i]] && this[childViews[i]].type === 'M.ButtonView') {
360                     var button = this[childViews[i]];
361                     if(button.isActive) {
362                         this.setActiveButton(button.id);
363                     }
364                 }
365             }
366         }
367     },
368 
369     /**
370      * This method returns the currently selected button of this button group. If no
371      * button is selected, null is returned.
372      *
373      * @returns {M.ButtonView} The currently active button of this button group.
374      */
375     getActiveButton: function() {
376         return this.activeButton;  
377     },
378 
379     /**
380      * This method activates one button within the button group.
381      *
382      * @param {M.ButtonView, String} id The button to be set active or its id.
383      */
384     setActiveButton: function(id, event, nextEvent) {
385         if(this.isSelectable) {
386             if(this.activeButton) {
387                 this.activeButton.removeCssClass('ui-btn-active');
388                 this.activeButton.isActive = NO;
389             }
390 
391             var button = M.ViewManager.getViewById(id);
392             if(!button) {
393                 if(id && typeof(id) === 'object' && id.type === 'M.ButtonView') {
394                     button = id;
395                 }
396             }
397             if(button) {
398                 button.addCssClass('ui-btn-active');
399                 button.isActive = YES;
400                 this.activeButton = button;
401             }
402         }
403 
404         /* trigger change event for the button group */
405         $('#' + this.id).trigger('change');
406 
407         /* delegate event to external handler, if specified */
408         if(nextEvent) {
409             M.EventDispatcher.callHandler(nextEvent, event, YES);
410         }
411     },
412 
413     /**
414      * Applies some style-attributes to the button group.
415      *
416      * @private
417      * @returns {String} The button group's styling as html representation.
418      */
419     style: function() {
420         var html = '';
421         if(this.numberOfLines && this.isInset) {
422             html += ' class="ui-listview';
423         }
424         if(this.cssClass) {
425             html += html !== '' ? ' ' + this.cssClass : ' class="' + this.cssClass;
426         }
427         html += '"';
428         return html;
429     }

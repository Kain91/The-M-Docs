 11 m_require('core/foundation/view_manager.js');
 12 
 13 /**
 14  * @class
 15  *
 16  * The root class for an application.
 17  *
 18  * @extends M.Object
 19  */
 20 M.Application = M.Object.extend(
 21 /** @scope M.Application.prototype */ {
 22 
 23     /**
 24      * The type of this object.
 25      *
 26      * @type String
 27      */
 28     type: 'M.Application',
 29 
 30     /**
 31      * The application's name.
 32      *
 33      * @type String
 34      */
 35     name: null,
 36 
 37     /**
 38      * The application's view manager.
 39      *
 40      * @type Object
 41      */
 42     viewManager: M.ViewManager,
 43 
 44     /**
 45      * The application's model registry.
 46      *
 47      * @type Object
 48      */
 49     modelRegistry: M.ModelRegistry,
 50 
 51     /**
 52      * The application's event dispatcher.
 53      *
 54      * @type Object
 55      */
 56     eventDispatcher: M.EventDispatcher,
 57 
 58     /**
 59      * The application's cypher object, used for encoding and decoding.
 60      *
 61      * @type Object
 62      */
 63     cypher: M.Cypher,
 64 
 65     /**
 66      * The application's current language.
 67      *
 68      * @type String
 69      */
 70     currentLanguage: null,
 71 
 72     /**
 73      * The application's default / fallback language.
 74      *
 75      * @type String
 76      */
 77     defaultLanguage: null,
 78 
 79     /**
 80      * This property determines whether to use transitions within the application, e.g. on
 81      * page switches, or not. If set to NO, there will be no framework-sided transitions.
 82      *
 83      * @type Boolean
 84      */
 85     useTransitions: YES,
 86 
 87     /**
 88      * This property is set to NO once the first page within an application was loaded. So this
 89      * can be used as a hook to trigger some actions at the first load of any view. To do initial
 90      * things for a specific view, use the isFirstLoad property of M.PageView.
 91      *
 92      * @type Boolean
 93      */
 94     isFirstLoad: YES,
 95 
 96     /**
 97      * This property can be used to define the application's entry page. If set, this page will
 98      * be the first to be displayed if your application is started.
 99      *
100      * Even if this property is not absolutely necessary, we highly recommend to specify an entry
101      * page! 
102      *
103      * @type String
104      */
105     entryPage: null,
106 
107     /**
108      * This property is used to store the application's title.
109      *
110      * @type String
111      */
112     applicationTitle: '',
113 
114     /**
115      * This method encapsulates the 'include' method of M.Object for better reading of code syntax.
116      * Basically it integrates the defined pages within the application into M.Application and sets
117      * some basic configuration properties, e.g. the default language.
118      *
119      * @param {Object} obj The mixed in object for the extend call.
120      */
121     design: function(obj) {
122         var pages = {};
123         for(var pageName in obj) {
124             if(obj[pageName] && obj[pageName].type === 'M.PageView') {
125                 pages[pageName] = obj[pageName];
126             }
127         }      
128         this.include({
129             pages: pages
130         });
131 
132         this.entryPage = ((obj.entryPage && typeof(obj.entryPage) === 'string') ? obj.entryPage : null);
133 
134         return this;
135     },
136 
137     /**
138      * The application's main-method, that is called automatically on load of the app.
139      * Inside this method the rendering is initiated and all pages are bound to the 'pageshow'
140      * event so one can do some action whenever a page is loaded.
141      */
142     main: function() {
143         var that = this;
144 
145         /* get the application's title */
146         this.applicationTitle = document.title;
147 
148         /* first lets get the entry page and remove it from pagelist and viewlist */
149         var entryPage = M.ViewManager.getPage(M.Application.entryPage);
150         delete this.viewManager.viewList[entryPage.id];
151         delete this.viewManager.pageList[entryPage.id];
152 
153         /* set the default id 'm_entryPage' for entry page */
154         entryPage.id = 'm_entryPage';
155 
156         /* now lets render entry page to get it into the DOM first and set it as the current page */
157         entryPage.render();
158         this.viewManager.setCurrentPage(entryPage);
159 
160         /* now lets render all other pages */
161         _.each(this.viewManager.pageList, function(page) {
162             page.render();
163         });
164 
165         /* finally add entry page back to pagelist and view list, but with new key 'm_entryPage' */
166         this.viewManager.viewList['m_entryPage'] = entryPage;
167         this.viewManager.pageList['m_entryPage'] = entryPage;
168     }
169 
170 });

112     /**
113      * Creates a new record of the model, means an instance of the model based on the blueprint.
114      * You pass the object's specific attributes to it as an object.
115      *
116      * @param {Object} obj The properties object, e.g. {firstname: 'peter', lastname ='fox'}
117      * @returns {Object} The model record with the passed properties set. State depends on newly creation or fetch from storage: if
118      * from storage then state is M.STATE_NEW or 'state_new', if fetched from database then it is M.STATE_VALID or 'state_valid'
119      */
120     createRecord: function(obj) {
121 
122         var rec = this.extend({
123             m_id: obj.m_id ? obj.m_id : M.Application.modelRegistry.getNextId(this.name),
124             record: obj /* properties that are added to record here, but are not part of __meta, are deleted later (see below) */
125         });
126         delete obj.m_id;
127         rec.state = obj.state ? obj.state : M.STATE_NEW;
128         delete obj.state;
129 
130         /* set timestamps if new */
131         if(rec.state === M.STATE_NEW) {
132             rec.record[M.META_CREATED_AT] = M.Date.now().format('yyyy/mm/dd HH:MM:ss');
133             rec.record[M.META_UPDATED_AT] = M.Date.now().format('yyyy/mm/dd HH:MM:ss');
134         }
135 
136         for(var i in rec.record) {
137 
138             if(i === 'ID' || i === M.META_CREATED_AT || i === M.META_UPDATED_AT) {
139                 continue;
140             }
141 
142             /* if record contains properties that are not part of __meta (means that are not defined in the model blueprint) delete them */
143             if(!rec.__meta.hasOwnProperty(i)) {
144                 delete rec.record[i];
145                 continue;
146             }
147 
148             /* if reference to a record entity is in param obj, assign it like in set. */
149             if(rec.__meta[i].dataType === 'Reference' && rec.record[i] && rec.record[i].type && rec.record[i].type === 'M.Model') {
150                 // call set of model
151                 rec.set(i, rec.record[i]);
152             }
153 
154             if(rec.__meta[i]) {
155                 rec.__meta[i].isUpdated = NO;
156             }
157         }
158 
159         this.recordManager.add(rec);
160         return rec;
161     },

163     /**
164      * Create defines a new model blueprint. It is passed an object with the model's attributes and the model's business logic
165      * and after it the type of data provider to use.
166      *
167      * @param {Object} obj An object defining the model's
168      * @param {Object} dp The data provider to use, e. g. M.LocalStorageProvider
169      * @returns {Object} The model blueprint: acts as blueprint to all records created with @link M.Model#createRecord
170      */
171     create: function(obj, dp) {
172         var model = M.Model.extend({
173             __meta: {},
174             name: obj.__name__,
175             dataProvider: dp,
176             recordManager: {},
177             usesValidation: obj.usesValidation === null || obj.usesValidation === undefined ? this.usesValidation : obj.usesValidation
178         });
179         delete obj.__name__;
180         delete obj.usesValidation;
181 
182         for(var prop in obj) {
183             if(typeof(obj[prop]) === 'function') {
184                 model[prop] = obj[prop];
185             } else if(obj[prop].type === 'M.ModelAttribute') {
186                 model.__meta[prop] = obj[prop];
187             }
188         }
189 
190         /* add ID, _createdAt and _modifiedAt properties in meta for timestamps  */
191         model.__meta['ID'] = this.attr('Integer', {
192             isRequired:NO
193         });
194         model.__meta[M.META_CREATED_AT] = this.attr('String', { // could be 'Date', too
195             isRequired:YES
196         });
197         model.__meta[M.META_UPDATED_AT] = this.attr('String', { // could be 'Date', too
198             isRequired:YES
199         });
200 
201         /* CouchDB documents have a rev property for managing versions*/
202         if(model.dataProvider.type === 'M.DataProviderCouchDb') {
203             model.__meta['rev'] = this.attr('String', {
204                 isRequired:NO
205             });
206         }
207 
208         model.recordManager = M.RecordManager.extend({records:[]});
209 
210         /* if dataprovider is WebSqlProvider, create table for this model and add ID ModelAttribute Object to __meta */
211         if(model.dataProvider.type === 'M.DataProviderWebSql') {
212             model.dataProvider.init({model: model, onError:function(err){M.Logger.log(err, M.ERR);}}, function() {});
213             model.dataProvider.isInitialized = YES;
214         }
215 
216         M.Application.modelRegistry.register(model.name);
217 
218         /* save model in modelList with model name as key */
219         this.modelList[model.name] = model;
220 
221         /* Re-set the just registered model's id, if there is a value stored */
222         /* Model Registry stores the current id of a model type into localStorage */
223         var m_id = localStorage.getItem(M.LOCAL_STORAGE_PREFIX + M.Application.name + M.LOCAL_STORAGE_SUFFIX + model.name);
224         if(m_id) {
225             M.Application.modelRegistry.setId(model.name, parseInt(m_id));
226         }
227         return model;
228     },

274     /**
275      * Get attribute propName from model, if async provider is used, get on references does not return the property value but a boolean indicating
276      * the load status. get must then be called again in onSuccess callback to retrieve the value
277      * @param {String} propName the name of the property whose value shall be returned
278      * @param {Object} obj optional parameter containing the load force flag and callbacks, e.g.:
279      * {
280      *   force: YES,
281      *   onSuccess: function() { console.log('yeah'); }
282      * }
283      * @returns {Boolean|Object|String} value of property or boolean indicating the load status
284      */
285     get: function(propName, obj) {
286         var metaProp = this.__meta[propName];
287         var recProp = this.record[propName];
288         /* return ref entity if property is a reference */
289         if(metaProp && metaProp.dataType === 'Reference') {
290             if(metaProp.refEntity) {// if entity is already loaded and assigned here in model record
291                 return metaProp.refEntity;
292             } else if(recProp) { // if model record has a reference set, but it is not loaded yet
293                 if(obj && obj.force) { // if force flag was set
294                     /* custom call to deepFind with record passed only being the one property that needs to be filled, type of dp checked in deepFind */
295                     var callback = this.dataProvider.isAsync ? obj.onSuccess : null
296                     this.deepFind([{
297                         prop: propName,
298                         name: metaProp.reference,
299                         model: this.modelList[metaProp.reference],
300                         m_id: recProp
301                     }], callback);
302                     if(!this.dataProvider.isAsync) { // if data provider acts synchronous, we can now return the fetched entity
303                         return metaProp.refEntity;
304                     }
305                     return YES;
306                 } else { // if force flag was not set, and object is not in memory and record manager load is not done and we return NO
307                     var r = this.recordManager.getRecordForId(recProp);
308                     if(r) { /* if object is already loaded and in record manager don't access storage */
309                         return r;
310                     } else {
311                         return NO; // return
312                     }
313                 }
314             } else { // if reference has not been set yet
315                 return null;
316             }
317         }
318         /* if propName is not a reference, but a "simple" property, just return it */
319         return recProp;
320     },

322     /**
323      * Set attribute propName of model with value val, sets' property to isUpdated (=> will be included in UPDATE call)
324      * and sets a new timestamp to _updatedAt. Will not do anything, if newVal is the same as the current prop value.
325      * @param {String} propName the name of the property whose value shall be set
326      * @param {String|Object} val the new value
327      */
328     set: function(propName, val) {
329 
330         /* TODO: implement hasMany... */
331         /* TODO: evaluate whether to save m_id in record and entity reference in __meta or other way round */
332         if(this.__meta[propName].dataType === 'Reference' && val.type && val.type === 'M.Model') {    // reference set
333             /* first check if new value is passed */
334             if(this.record[propName] !== val.m_id) {
335                 /* set m_id of reference in record */
336                 this.record[propName] = val.m_id;
337                 this.__meta[propName].refEntity = val;
338             }
339             return;
340         }
341 
342         if(this.record[propName] !== val) {
343             this.record[propName] = val;
344             this.__meta[propName].isUpdated = YES;
345             /* mark record as updated with new timestamp*/
346             this.record[M.META_UPDATED_AT] = M.Date.now().format('yyyy/mm/dd HH:MM:ss');
347         }

360     /**
361      * Validates the model, means calling validate for each property.
362      * @returns {Boolean} Indicating whether this record is valid (YES|true) or not (NO|false).
363      */
364     validate: function() {
365         var isValid = YES;
366         var validationErrorOccured = NO;
367         /* clear validation error buffer before validation */
368         M.Validator.clearErrorBuffer();
369 
370         /*
371         * validationBasis depends on the state of the model: if the model is in state NEW, all properties (__meta includes all)
372         * shall be considered for validation. if model is in another state, the model's record is used. example: the model is loaded from
373         * a database with only two properties included (select name, age FROM...). record now only contains these two properties but __meta
374         * still has all properties listed. models are valid if loaded from database so when saved again only the loaded properties need to get
375         * validated because all others have not been touched. that's why then record is used.
376         * */
377         var validationBasis = this.state === M.STATE_NEW ? this.__meta : this.record;
378 
379         for (var i in validationBasis) {
380             if(i === 'ID') { // skip property ID
381                 continue;
382             }
383             var prop = this.__meta[i];
384             var obj = {
385                 value: this.record[i],
386                 modelId: this.name + '_' + this.m_id,
387                 property: i
388             };
389             if (!prop.validate(obj)) {
390                 isValid = NO;
391             }
392         }
393         /* set state of model */
394         /*if(!isValid) {
395             this.state = M.STATE_INVALID;
396         } else {
397             this.state = M.STATE_VALID;
398         }*/
399         return isValid;
400     },

404      * Calls the corresponding find() of the data provider to fetch data based on the passed query or key.
405      *
406      * @param {Object} obj The param object with query or key and callbacks.
407      * @returns {Boolean|Object} Depends on data provider used. When WebSQL used, a boolean is returned, the find result is returned asynchronously,

410     find: function(obj){
411         if(!this.dataProvider) {
412             M.Logger.log('No data provider given.', M.ERR);

414         obj = obj ? obj : {};
415         /* check if the record list shall be cleared (default) before new found model records are appended to the record list */
416         /* TODO: needs to be placed in callback */
417         obj['deleteRecordList'] = obj['deleteRecordList'] ? obj.deleteRecordList : YES;
418         if(obj.deleteRecordList) {
419             this.recordManager.removeAll();
420         }
421         if(!this.dataProvider) {
422             M.Logger.log('No data provider given.', M.ERR);
423         }
424 
425         /* extends the given obj with self as model property in obj */
426         return this.dataProvider.find( $.extend(obj, {model: this}) );

429     /**
430      * Create or update a record in storage if it is valid (first check this).
431      * If param obj includes cascade:YES then save is cascadaded through all references recursively.
432      *
433      * @param {Object} obj The param object with query, cascade flag and callbacks.
434      * @returns {Boolean} The result of the data provider function call. Is a boolean. With LocalStorage used, it indicates if the save operation was successful.
435      * When WebSQL is used, the result of the save operation returns asynchronously. The result then is just the standard result returned by the web sql provider's save method
436      * which does not necessarily indicate whether the operation was successful, because the operation is asynchronous, means the operation's result is not predictable.
437      */
438     save: function(obj) {
439         if(!this.dataProvider) {
440             M.Logger.log('No data provider given.', M.ERR);
441         }
442         obj = obj ? obj: {};
443         if(!this.m_id) {
444             return NO;
445         }
446         var isValid = YES;
447 
448         if(this.usesValidation) {
449             isValid = this.validate();
450         }
451 
452         if(obj.cascade) {
453             for(var prop in this.__meta) {
454                 if(this.__meta[prop] && this.__meta[prop].dataType === 'Reference' && this.__meta[prop].refEntity) {
455                     this.__meta[prop].refEntity.save({cascade:YES}); // cascades recursively through all referenced model records
456                 }
457             }
458         }
459 
460         if(isValid) {
461             return this.dataProvider.save($.extend(obj, {model: this}));
462         }
463     },

513      * completes the model record by loading all referenced entities.
514      *
515      * @param {Function | Object} obj The param object with query, cascade flag and callbacks.

517     complete: function(callback) {
518         //console.log('complete...');
519         var records = [];
520         for(var i in this.record) {
521             if(this.__meta[i].dataType === 'Reference') {
522                 //records.push(this.__meta[i].refEntity);
523                 records.push({
524                     prop:i,
525                     name: this.__meta[i].reference,
526                     model: this.modelList[this.__meta[i].reference],
527                     m_id: this.record[i]
528                 });
529             }
530         }
531         this.deepFind(records, callback);

534     // TODO: handle onSuccess AND onError
535     deepFind: function(records, callback) {
536         //console.log('deepFind...');
537         //console.log('### records.length: ' + records.length);
538         if(records.length < 1) {    // recursion end constraint
539             if(callback) {
540                 callback();
541             }
542             return;
543         }
544         var curRec = records.pop(); // delete last element, decreases length of records by 1 => important for recursion constraint above
545         var cb = this.bindToCaller(this, this.deepFind,[records, callback]); // cb is callback for find in data provider
546         var that = this;
547 
548 
549         switch(this.dataProvider.type) {
550 
551             case 'M.DataProviderWebSql':
552                 this.modelList[curRec.name].find({     // call find on to fetched model record object
553                     constraint: {
554                         statement: 'WHERE ' + M.META_M_ID + ' = ? ',
555                         parameters: [curRec.m_id] // length must match number of ? in statements
556                     },
557 
558                     onSuccess: function(result) {
559                         that.setReference(result, that, curRec.prop, cb);
560                     },
561                     onError: function(err) {
562                         M.Logger.log('Error: ' + err, M.ERR);
563                     }
564                 });
565 
566                 break;
567 
568             case 'M.DataProviderLocalStorage':
569 
570                 var ref = this.modelList[curRec.name].find({
571                     key: curRec.m_id
572                 });
573 
574                 this.__meta[curRec.prop].refEntity = ref;
575 
576                 this.deepFind(records, callback); // recursion
577                 break;
578 
579             default:
580 
581                 break;
582         }
583     },

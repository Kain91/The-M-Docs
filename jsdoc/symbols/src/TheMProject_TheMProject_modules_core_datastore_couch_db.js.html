 93     /**
 94      * Makes a call to the remote couchdb instance on url + '_all_dbs'. Checks if needed DB already exists by iterating
 95      * through response array. Calls {@link M.WebSqlProvider#createDatabase} if it does not exist, saved internal callback if it exists.
 96      * Delegates to {@link M.DataProvider#performRequest} to make the actual request.
 97      *
 98      * @param {Object} obj The param obj, includes model. Not used here, just passed through.
 99      * @private
100      */
101     check: function(obj) {
102         //console.log('check called...');
103         var dbName = this.config.dbName;
104         var url = this.buildUrl('/_all_dbs');  // http://mycouchdb.com/_all_bs returns all databases of myCouchDB
105 
106         var that = this;
107         if(obj.showLoader) {
108             M.LoaderView.show();
109         }
110 
111         this.performRequest('GET', url, YES, null, null,
112             function(dbs) { /* onSuccess */
113 
114                 if(obj.showLoader) {
115                     M.LoaderView.hide();
116                 }
117 
118                 /* check if db already exists by looking if dbName is part of returned array of DB names */
119                 if(_.include(dbs, dbName)) {
120                     M.Logger.log('DB: ' + dbName + ' exists', M.INFO);
121                     that.isInitialized = YES;
122                     that.internalCallback(obj);
123                 } else {
124                     M.Logger.log('DB not existant, must be created first...', M.INFO);
125                     that.isInitialized = NO;
126                     that.createDatabase(obj);
127                 }
128             },
129             function(xhr, msg) { /* onFailure */
130 
131                 if(obj.showLoader) {
132                     M.LoaderView.hide();
133                 }
134 
135                 M.Logger.log('Connection error...', M.ERR);
136                 var err = M.Error.extend({
137                     code: M.ERR_CONNECTION,
138                     msg: msg
139                 });
140                 that.errorCallback(obj, err);
141             }
142         );
143     },

153             if(obj.onError.target && obj.onError.action) {
154                 obj.onError = this.bindToCaller(obj.onError.target, obj.onError.target[obj.onError.action], error, obj, this);
155                 obj.onError();
156             } else if(typeof(obj.onError) === 'function') {
157                 obj.onError(error, obj, this);

219     /**
220      * Saves a document to the CouchDB database with a Unique Id generated by {@link M.UniqueId#uuid}.
221      * Distinguishes between create and update request based on state of model record.
222      * Delegates to {@link M.DataProvider#performRequest} to make the actual request.
223      * 
224      * @param {Object} obj Param obj, contains the model record
225      */
226     save: function(obj) {
227         //console.log('save called...');
228         if(!this.isInitialized) {
229             this.internalCallback = this.save;
230             this.init(obj);
231             return;
232         }
233 
234         var that = this;
235 
236         if(obj.model.state === M.STATE_NEW) { /* make create request */
237             /* create unique id as key */
238             var uuid = M.UniqueId.uuid(32);
239             /* note: uniqued id is first assigned to model record on successful request */
240 
241             var dataValue =  JSON.stringify(obj.model.record);
242             //M.Logger.log(dataValue, M.INFO);
243             var url = this.buildUrl('/' + this.config.dbName + '/' + uuid);
244             //console.log('save URL: ' + url);
245 
246             if(obj.showLoader) {
247                     M.LoaderView.hide();
248             }
249             this.performRequest('PUT', url, YES, dataValue,
250 
251                 function(xhr) {    /* beforeSend set HTTP header to PUT */
252                     xhr.setRequestHeader("X-Http-Method-Override", 'PUT');
253                 },
254 
255                 function(data) { /* onSuccess */
256 
257                     if(obj.showLoader) {
258                         M.LoaderView.hide();
259                     }
260 
261                     /* assign returned uuid to model record */
262                     if(data.ok) {
263                         //console.log('set record ID from response');
264                         obj.model.set('ID', data.id);
265                         obj.model.set('rev', data.rev);
266                         if(obj.onSuccess.target && obj.onSuccess.action) {
267                             // returns obj.model
268                             obj.onSuccess = that.bindToCaller(obj.onSuccess.target, obj.onSuccess.target[obj.onSuccess.action], null, obj, this);
269                             obj.onSuccess();
270                         } else if(obj.onSuccess && typeof(obj.onSuccess) === 'function') {
271                             obj.onSuccess(null, obj, this);
272                         }
273                     }
274                 },
275 
276                 function(xhr, msg) { /* onError */
277 
278                     if(obj.showLoader) {
279                         M.LoaderView.hide();
280                     }
281 
282                     var data = JSON.parse(xhr.responseText);
283 
284                     var err = {};
285 
286                     if(data.error) {
287                         err = that.buildErrorObject(data);
288                     } else {
289                         err = M.Error.extend({
290                             code: M.ERR_CONNECTION,
291                             msg: msg
292                         });
293                     }
294                     that.errorCallback(obj, err);
295                 }
296             )
297         } else { /* update request */
298 
299             var dataValue =  JSON.stringify(obj.model.record);
300             //M.Logger.log(dataValue, M.INFO);
301             var url = this.buildUrl('/' + this.config.dbName + '/' + obj.model.get('ID') + '?rev=' + obj.model.get('rev'));
302             //console.log('save URL: ' + url);
303 
304             if(obj.showLoader) {
305                 M.LoaderView.show();
306             }
307 
308             this.performRequest('PUT', url, YES, dataValue,
309                 function() {    /* beforeSend */
310                     xhr.setRequestHeader("X-Http-Method-Override", 'PUT');
311                 },
312                 function(data) {    /* onSuccess */
313 
314                     if(obj.showLoader) {
315                         M.LoaderView.hide();
316                     }
317 
318                     if(data.ok) {
319                          obj.model.set('rev', data.rev);
320                         if(obj.onSuccess.target && obj.onSuccess.action) {
321                             /* returns saved model. */
322                             obj.onSuccess = that.bindToCaller(obj.onSuccess.target, obj.onSuccess.target[obj.onSuccess.action], null, obj, this);
323                             obj.onSuccess();
324                         } else if(obj.onSuccess && typeof(obj.onSuccess) === 'function') {
325                             obj.onSuccess(null, obj, this);
326                         }
327                     }
328                 },
329                 function(xhr, msg) {    /* onError */
330 
331                     if(obj.showLoader) {
332                         M.LoaderView.hide();
333                     }
334 
335                     var data = JSON.parse(xhr.responseText);
336 
337                     var err = {};
338 
339                     if(data.error) {
340                         err = that.buildErrorObject(data);
341                     } else {
342                         err = M.Error.extend({
343                             code: M.ERR_CONNECTION,
344                             msg: msg
345                         });
346                     }
347                     that.errorCallback(obj, err);
348                 }
349             );

430     /**
431      *
432      * Finds all Documents of a database by first retrieving the list of all documents and then making GET requests
433      * for each document in the list.
434      * findAllDocuments is called by itself recursively until callsLeft is less than one. (callsLeft is decremented in each call)
435      * Delegates to {@link M.DataProvider#performRequest} to make the actual request.
436      *
437      * @param {Object} obj Param obj
438      * @param {Boolean} first Determines whether it is first call (that retrieves all documents) or later calls (each GET request)
439      * @param {Array} docList List of documents. Each member is fetched via GET request
440      * @param {Array} result Results array: each response of a request is saved in this result array
441      * @param {Number} callsLeft Counter indicating how many calls are left for retrieving all documents. if less than zero function returns.
442      * @private
443      */
444     findAllDocuments: function(obj, isFirst, docList, result, callsLeft) {
445         //M.Logger.log('findAllDocuments called...', M.INFO);
446         /*if(!this.isInitialized) {
447             console.log('not initialized in findAllDocuments...');
448             this.internalCallback = this.findAllDocuments;
449             this.init(obj);
450             return;
451         }*/
452 
453         var that = this;
454 
455         if(isFirst) {
456             //console.log('findAllDocuments: isFirst');
457             var url = this.buildUrl('/' + this.config.dbName + '/_all_docs');
458 
459             if(obj.showLoader) {
460                 M.LoaderView.show();
461             }
462 
463             this.performRequest('GET', url, YES, null, null,
464                 function(data) {  // onSuccess callback
465 
466                     if(obj.showLoader) {
467                         M.LoaderView.hide();
468                     }
469 
470                     /*
471                     * result:
472                     * {"total_rows":10,"offset":0,"rows":[
473                         {"id":"123456","key":"123456","value":{"rev":"1-014e3877e6395dea9415867442d41d4d"}},
474                         ...
475                         ]
476                       }
477                     **/
478                     that.findAllDocuments(obj, NO, data.rows, [], data.total_rows);
479                 },
480                 function(xhr, msg) { // onError callback
481 
482                     if(obj.showLoader) {
483                         M.LoaderView.hide();
484                     }
485 
486                     var err = M.Error.extend({
487                         code: M.ERR_CONNECTION,
488                         msg: msg
489                     });
490                     that.errorCallback(obj, err);
491                 }
492             );
493             return;
494         } else { /* if second, get all documents one by one and return them */
495             // if request fails for one, it fails completely and therefor reset records array in recordmanager completely
496             // TODO: is this the desired behaviour?
497             if(callsLeft > 0) {
498                 var url = this.buildUrl('/' + this.config.dbName + '/' + docList[callsLeft - 1].id); // -1 because we use length as index, which is one bigger than index
499 
500                 if(obj.showLoader) {
501                     M.LoaderView.show();
502                 }
503 
504                 this.performRequest('GET', url, YES, null, null,
505                     function(data) { /* onSuccess */
506 
507                         if(obj.showLoader) {
508                             M.LoaderView.hide();
509                         }
510 
511                         if(!data.error) {
512                             result.push(that.createRecOfDoc(obj, data));
513                             callsLeft = callsLeft - 1;
514                             that.findAllDocuments(obj, NO, docList, result, callsLeft);
515                         }
516                     },
517                     function(xhr, msg) { /* onError */
518 
519                         if(obj.showLoader) {
520                             M.LoaderView.hide();
521                         }
522 
523                         var data = JSON.parse(xhr.responseText);
524                         var err = {};
525                         if(data.error) {
526                             err = that.buildErrorObject(data);
527                         } else {
528                             err = M.Error.extend({
529                                 code: M.ERR_CONNECTION,
530                                 msg: msg
531                             });
532                         }
533                         obj.model.recordManager.removeAll();
534                         that.errorCallback(obj, err);
535                         return;
536                     }
537                 );
538 
539             } else {
540                 if(obj.onSuccess && obj.onSuccess.target && obj.onSuccess.action) {
541                     /* [result] is a workaround for bindToCaller: bindToCaller uses call() instead of apply() if an array is given.
542                      * result is an array, but what call is doing with it is wrong in this case. call maps each array element to one method
543                      * parameter of the function called. Our callback only has one parameter so it would just pass the first value of our result set to the
544                      * callback. therefor we now put result into an array (so we have an array inside an array) and result as a whole is now passed as the first
545                      * value to the callback.
546                      *  */
547                     obj.onSuccess = that.bindToCaller(obj.onSuccess.target, obj.onSuccess.target[obj.onSuccess.action], [result], obj, this);
548                     obj.onSuccess();
549                 } else if(obj.onSuccess && typeof(obj.onSuccess) === 'function') {
550                     obj.onSuccess(result, obj, this);
551                 }
552                 // make onSuccessCallback
553                 return result;
554             }
555         }

565     /**
566      * Creates a model record from the response
567      * @param obj The param obj
568      * @param data the data response from the server
569      * @returns {Object} model record
570      * @private
571      */
572     createRecOfDoc: function(obj, data) {
573         //var rec = JSON.parse(data);
574         var rec = data;
575         rec['ID'] = rec['_id'];
576         rec['rev'] = rec['_rev'];
577 
578         var myRec = obj.model.createRecord($.extend(rec, {state: M.STATE_VALID}));
579 
580         /* create M.Date objects for all date properties */
581         for(var j in myRec.__meta) {
582             /* here we can work with setter and getter because myRec already is a model record */
583             if(myRec.__meta[j].dataType === 'Date' && typeof(myRec.get(j)) === 'string') {
584                 myRec.set(j, M.Date.create(myRec.get(j)));
585             }
586         }
587         return myRec;
588     },

713     /**
714      * Wrapper function to perform the actual request with {@link M.Request}.
715      * @param {String} method The HTTP method to use
716      * @param {String} url The request URL
717      * @param {Boolean} isJson Flag that determines whether request shall be proceeded as JSON
718      * @param {Object|String} data The HTTP-Body
719      * @param {Function} beforeSend Function that is called before request is sent
720      * @param {Function} onSuccess Success callback function
721      * @param {Function} onError Error callback function

737     /**
738      * Constructs a {@link M.Error} object.
739      * @param {Object} resp The response object, containg additional error information like 'reason' and 'error' type
740      * @returns {Object} The error object
741      */
742     buildErrorObject: function(resp) {
743         var err = M.Error.extend({
744             msg: resp.reason
745         });
746 
747         switch(resp.error) {    // switch through possible errors returned by CouchDB and add related error to error obj
748             case 'conflict':
749                 err.code = M.ERR_COUCHDB_CONFLICT;
750                 break;
751             case 'not_found':
752                 if(resp.reason === 'missing') {
753                     err.code = M.ERR_COUCHDB_DOCNOTFOUND;
754                 } else if (resp.reason === 'no_db_file') {
755                     err.code = M.ERR_COUCHDB_DBNOTFOUND;
756                 }
757                 break;
758             case 'file_exists':
759                 err.code = M.ERR_COUCHDB_DBEXISTS;
760                 break;
761             default:
762                 err.code = M.ERR_UNDEFINED
763                 break;
764         }
765         return err;
766     },

795     /**
796      * Makes given DB name CouchDB compatible. Rules of CouchDB DB name:
797      * "Only lowercase characters (a-z), digits (0-9), and any of the characters _, $, (, ), +, -, and / are allowed. Must begin with a letter."
798      * @param {String} name The sanitized, CouchDB compatible, DB name
799      * @private
800      */
801     sanitizeDbName: function(name) {
802         var origName = name;
803         /* only lowercase letters allowed */
804         name = name.toLowerCase();
805         /* dot not allowed in name, replace with _ , whitelist below doesn't match dot  */
806         name = name.replace(/\./g, "_");
807         /* replace all other illegal characters (all others thant the one mentioned below) with _ */
808         name = name.replace(/[^a-z0-9_$()+-/]/g, "_");
809         /* must begin with a letter: delete all prefixes that are not letters */
810         name = name.replace(/^[^a-z]+/, "");
811         /* what if only numbers are given? */
812 
813         /* Log a warning to inform user about the renaming */
814         if(origName !== name) {
815             M.Logger.log('The given db name: ' + origName + ' is not a valid CouchDB name. It has been sanitized to: ' + name, M.WARN);
816         }
817         return name;
818     }

208     /**
209      * Renders a map view, respectively a map view container.
210      *
211      * @private
212      * @returns {String} The map view's html representation.
213      */
214     render: function() {
215         this.html += '<div data-fullscreen="true" id="' + this.id + '"';
216         this.html += !this.isInset ? ' class="ui-listview"' : '';
217         this.html += '><div id="' + this.id + '_map"' + this.style() + '></div></div>';
218 
219         return this.html;
220     },
221 
222     /**
223      * This method is called if the bound content changed. This content must be
224      * an array of M.Location objects or M.MapMarkerView objects. This method
225      * will take care of a re-rendering of the map view and all of its bound
226      * markers.
227      *
228      * If M.Location objects are passed, the default settings for map markers
229      * of this map view are assigned.
230      *
231      * Note that you can not use individual click events for your markers if
232      * you pass M.Location objects.
233      */
234     renderUpdate: function() {
235         /* check if content binding is valid */
236         var content = null;
237         if(!(this.contentBinding && this.contentBinding.target && typeof(this.contentBinding.target) === 'object' && this.contentBinding.property && typeof(this.contentBinding.property) === 'string' && this.contentBinding.target[this.contentBinding.property])) {
238             M.Logger.log('No valid content binding specified for M.MapView (' + this.id + ')!', M.WARN);
239             return;
240         }
241 
242         /* get the marker / location objects from content binding */
243         var content = this.contentBinding.target[this.contentBinding.property];
244         var markers = [];
245         
246         /* save a reference to the map */
247         var that = this;
248 
249         /* if we got locations, transform to markers */
250         if(content && content[0] && content[0].type === 'M.Location') {
251             _.each(content, function(location) {
252                 if(location && typeof(location) === 'object' && location.type === 'M.Location') {
253                     markers.push(M.MapMarkerView.init({
254                         location: location,
255                         map: that
256                     }));
257                 }
258             });
259         /* otherwise check and filter for map markers */
260         } else if(content && content[0] && content[0].type === 'M.MapMarkerView') {
261             markers = _.select(content, function(marker) {
262                 return (marker && marker.type === 'M.MapMarkerView')
263             })
264         }
265 
266         /* remove current markers */
267         if(this.removeMarkersOnUpdate) {
268             this.removeAllMarkers();
269         }
270 
271         /* add all new markers */
272         _.each(markers, function(marker) {
273             that.addMarker(marker);
274         })
275     },

303     /**
304      * This method is used to initialize a map view, typically out of a controller.
305      * With its options parameter you can set or update almost every parameter of
306      * a map view. This allows you to define a map view within your view, but then
307      * update its parameters later when you want this view to display a map.
308      *
309      * The options parameter must be passed as a simple object, containing all of
310      * the M.MapView's properties you want to be updated. Such an options object
311      * could look like the following:
312      *
313      *   {
314      *     zoomLevel: 12,
315      *     mapType: M.MAP_HYBRID,
316      *     initialLocation: location
317      *   }
318      *
319      * While all properties of the options parameter can be given as Number, String
320      * or a constant value, the location must be a valid M.Location object.
321      *
322      * @param {Object} options The options for the map view.
323      */
324     initMap: function(options, isUpdate) {
325         if(!this.isInitialized || isUpdate) {
326             if(!isUpdate) {
327                 this.markers = [];
328             }
329             for(var i in options) {
330                  switch (i) {
331                      case 'zoomLevel':
332                         this[i] = (typeof(options[i]) === 'number' && options[i] > 0) ? (options[i] > 22 ? 22 : options[i]) : this[i];
333                         break;
334                      case 'mapType':
335                         this[i] = (options[i] === M.MAP_ROADMAP || options[i] === M.MAP_HYBRID || options[i] === M.MAP_SATELLITE || options[i] === M.MAP_TERRAIN) ? options[i] : this[i];
336                         break;
337                      case 'markerAnimationType':
338                         this[i] = (options[i] === M.MAP_MARKER_ANIMATION_BOUNCE || options[i] === M.MAP_MARKER_ANIMATION_DROP) ? options[i] : this[i];
339                         break;
340                      case 'showMapTypeControl':
341                      case 'showNavigationControl':
342                      case 'showStreetViewControl':
343                      case 'isDraggable':
344                      case 'setMarkerAtInitialLocation':
345                      case 'removeMarkersOnUpdate':
346                         this[i] = typeof(options[i]) === 'boolean' ? options[i] : this[i];
347                         break;
348                      case 'initialLocation':
349                         this[i] = (typeof(options[i]) === 'object' && options[i].type === 'M.Location') ? options[i] : this[i];
350                         break;
351                      default:
352                         break;
353                  }
354             };
355             if(isUpdate) {
356                 if(this.removeMarkersOnUpdate) {
357                     this.removeAllMarkers();
358                 }
359                 this.map.setOptions({
360                     zoom: this.zoomLevel,
361                     center: new google.maps.LatLng(this.initialLocation.latitude, this.initialLocation.longitude),
362                     mapTypeId: google.maps.MapTypeId[this.mapType],
363                     mapTypeControl: this.showMapTypeControl,
364                     navigationControl: this.showNavigationControl,
365                     streetViewControl: this.showStreetViewControl,
366                     draggable: this.isDraggable
367                 });
368             } else {
369                 this.map = new google.maps.Map($('#' + this.id + '_map')[0], {
370                     zoom: this.zoomLevel,
371                     center: new google.maps.LatLng(this.initialLocation.latitude, this.initialLocation.longitude),
372                     mapTypeId: google.maps.MapTypeId[this.mapType],
373                     mapTypeControl: this.showMapTypeControl,
374                     navigationControl: this.showNavigationControl,
375                     streetViewControl: this.showStreetViewControl,
376                     draggable: this.isDraggable
377                 });
378             }
379 
380             if(this.setMarkerAtInitialLocation) {
381                 var that = this;
382                 this.addMarker(M.MapMarkerView.init({
383                     location: this.initialLocation,
384                     map: that.map
385                 }));
386             }
387             
388             this.isInitialized = YES;
389         } else {
390             M.Logger.log('The M.MapView has already been initialized', M.WARN);
391         }
392     },

394     /**
395      * This method is used to update a map view, typically out of a controller.
396      * With its options parameter you can update or update almost every parameter
397      * of a map view. This allows you to define a map view within your view, but
398      * then update its parameters later when you want this view to display a map
399      * and to update those options over and over again for this map. 
400      *
401      * The options parameter must be passed as a simple object, containing all of
402      * the M.MapView's properties you want to be updated. Such an options object
403      * could look like the following:
404      *
405      *   {
406      *     zoomLevel: 12,
407      *     mapType: M.MAP_HYBRID,
408      *     initialLocation: location
409      *   }
410      *
411      * While all properties of the options parameter can be given as Number, String
412      * or a constant value, the location must be a valid M.Location object.
413      *
414      * @param {Object} options The options for the map view.
415      */
416     updateMap: function(options) {
417         this.initMap(options, YES);
418     },
419 
420     /**
421      * This method can be used to add a marker to the map view. Simply pass a
422      * valid M.MapMarkerView object and a map marker is created automatically,
423      * displayed on the map and added to this map view's markers property.
424      *
425      * @param {M.MapMarkerView} marker The marker to be added.
426      */
427     addMarker: function(marker) {
428         if(marker && typeof(marker) === 'object' && marker.type === 'M.MapMarkerView') {
429             var that = this;
430             marker.marker = new google.maps.Marker({
431                 map: that.map,
432                 draggable: NO,
433                 animation: google.maps.Animation[marker.markerAnimationType ? marker.markerAnimationType : that.markerAnimationType],
434                 position: new google.maps.LatLng(marker.location.latitude, marker.location.longitude)
435             });
436             marker.registerEvents();
437             this.markers.push(
438                 marker
439             );
440         } else {
441             M.Logger.log('No valid M.MapMarkerView passed for addMarker().', M.WARN);
442         }
443     },
444 
445     /**
446      * This method can be used to remove a certain marker from the map view. In
447      * order to do this, you need to pass the M.MapMarkerView object that you
448      * want to be removed from the map view.
449      *
450      * @param {M.MapMarkerView} marker The marker to be removed.
451      */
452     removeMarker: function(marker) {
453         if(marker && typeof(marker) === 'object' && marker.type === 'M.MapMarkerView') {
454             var didRemoveMarker = NO;
455             this.markers = _.select(this.markers, function(m) {
456                 if(marker === m){
457                     m.marker.setMap(null);
458                     didRemoveMarker = YES;
459                 }
460                 return !(marker === m);
461             });
462             if(!didRemoveMarker) {
463                 M.Logger.log('No marker found matching the passed marker in removeMarker().', M.WARN);    
464             }
465         } else {
466             M.Logger.log('No valid M.MapMarkerView passed for removeMarker().', M.WARN);
467         }
468     },
469 
470     /**
471      * This method removes all markers from this map view. It both cleans up the
472      * markers array and deletes the marker's visual representation from the map
473      * view.
474      */
475     removeAllMarkers: function() {
476         _.each(this.markers, function(marker) {
477             marker.marker.setMap(null);
478         });
479         this.markers = [];
480     }

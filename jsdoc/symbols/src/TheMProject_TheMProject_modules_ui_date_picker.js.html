 11 /**
 12  * @class
 13  *
 14  * This defines the prototype for a date picker view.
 15  *
 16  * @extends M.View
 17  */
 18 M.DatePickerView = M.View.extend(
 19 /** @scope M.DatePickerView.prototype */ {
 20 
 21     /**
 22      * The type of this object.
 23      *
 24      * @type String
 25      */
 26     type: 'M.DatePickerView',
 27 
 28     /**
 29      * This property specifies whether a value update is currently in progress
 30      * or not. This is mainly used to decide on focus event for the text field
 31      * whether or not to show the date picker view.
 32      *
 33      * @type Boolean
 34      */
 35     valueUpdateInProgress: NO,
 36 
 37     /**
 38      * Determines whether this date picker is also used to pick a time. If set
 39      * to YES, two input fields (text fields) are rendered below the date picker
 40      * to allow the user to enter a certain time (hours and minutes).
 41      *
 42      * @type Boolean
 43      */
 44     isDateTimePicker: YES,
 45 
 46     /**
 47      * Renders a datepicker as a button. On click, there will be a date picker
 48      * visible to select a certain date.
 49      *
 50      * @private
 51      * @returns {String} The date picker view's html representation.
 52      */
 53     render: function() {
 54         this.html += '<input type="hidden" id="' + this.id + '_hidden" />';
 55         this.html += '<a data-role="button" style="display:none;" data-type="date" href="#" id="' + this.id + '"' + this.style() + '>-</a>';
 56 
 57         if(this.isDateTimePicker) {
 58             this.html += '<div id="' + this.id + '_time" class="ui-grid-a">';
 59             this.html += '<div class="ui-block-a">';
 60             this.html += '<input type="number" id="' + this.id + '_hours" onchange="M.ViewManager.getViewById(\'' + this.id + '\').dateSelected();" value="00" />';
 61             this.html += '</div>';
 62             this.html += '<div class="ui-block-b">';
 63             this.html += '<input type="number" id="' + this.id + '_minutes" onchange="M.ViewManager.getViewById(\'' + this.id + '\').dateSelected();" value="00" />';
 64             this.html += '</div>';
 65             this.html += '</div>';
 66         }
 67 
 68         /* set the datepicker's language to the user's language (if specified) */
 69         var language = M.I18N.getLanguage();
 70         if($.datepicker.regional[language]) {
 71             $.datepicker.setDefaults(language);
 72         } else if($.datepicker.regional[language.substring(0, 2)]) {
 73             $.datepicker.setDefaults($.datepicker.regional[language.substring(0, 2)]);
 74         }
 75 
 76         this.internalTarget = this;
 77         this.internalAction = 'showDatePicker';
 78         return this.html;
 79     },
 80 
 81     /**
 82      * Updates a date picker view based on its newly set value.
 83      *
 84      * @private
 85      */
 86     renderUpdate: function() {
 87         $('#' + this.id + '_hours').val(this.value.format('HH'));
 88         $('#' + this.id + '_minutes').val(this.value.format('MM'));
 89         $('#' + this.id + ' ~ div:first').datepicker('setDate', this.value.date);
 90         $('#' + this.id + ' span.ui-btn-text').html($('#' + this.id + '_hidden').val());
 91 
 92         this.computeDate();
 93     },
 94 
 95     /**
 96      * Triggers the rendering engine, jQuery mobile, to style the date picker.
 97      *
 98      * @private
 99      */
100     theme: function() {
101         $('.hasDatepicker').hide();
102         $('#' + this.id + ' span.ui-btn-text').html($('#' + this.id + '_hidden').val());
103         this.computeDate();
104     },
105 
106     /**
107      * This method computes the date out of the selected date and a given time.
108      */
109     computeDate: function() {
110         var date = M.Date.create($('#' + this.id + ' ~ div:first').datepicker("getDate"));
111         var hours = parseInt($('#' + this.id + '_hours').val());
112         var minutes = parseInt($('#' + this.id + '_minutes').val());
113 
114         hours = (isNaN(hours) || hours < 0 || hours > 23) ? 0 : hours;
115         minutes = (isNaN(minutes) || minutes < 0 || minutes > 59) ? 0 : minutes;
116 
117         this.value = date.hoursFromDate(hours).minutesFromDate(minutes);
118 
119         //TODO: build generic 
120         this.bindToCaller(this.target, this.target[this.action], this.value)();
121     },
122 
123     /**
124      * This method is called whenever the button of a date picker view is clicked. 
125      * We use this event to show the actual date picker and to let the user select
126      * a date. Once the user clicks a date, the date picker gets hidden again and
127      * we update the button's label.
128      */
129     showDatePicker: function() {
130         if(this.valueUpdateInProgress) {
131             return;
132         }
133 
134         $('.hasDatepicker').show();
135 
136         var that = this;
137         $('#' + this.id + ' ~ div.hasDatepicker table.ui-datepicker-calendar').live(
138             'mouseup',
139             function() {
140                 $('.hasDatepicker').hide();
141                 that.removeCssClass('ui-btn-active');
142             }
143         );
144     },
145 
146     /**
147      * This method is event triggered and gets called automatically if a user selected
148      * a certain date. We then update the button's label and the date picker's value.
149      *
150      * @param {String} dateText The date string to be displayed as the button's label.
151      */
152     dateSelected: function(dateText) {
153         $('#' + this.id + ' span.ui-btn-text').html(dateText);
154         this.computeDate();
155     },
156 
157     /**
158      * Set the value, respectively the date, of this date picker view.
159      *
160      * @param {M.Date} date The date, the date picker should be set to.
161      */
162 
163     setDate: function(date) {
164         $('#' + this.id + ' ~ div:first').datepicker('setDate', date.date);
165         this.value = M.Date.create($('#' + this.id + ' ~ div:first').datepicker("getDate"));
166     },
167 
168     /**
169      * Get the value, respectively the date, of this date picker view.
170      *
171      * @returns {M.Date} The date of this date picker.
172      */
173 
174     getDate: function() {
175         return this.value;
176     },
177 
178     /**
179      * This method sets its value to the value it has in its DOM representation
180      * and then delegates these changes to a controller property if the
181      * contentBindingReverse property is set.
182      *
183      * Additionally call target / action if set.
184      *
185      * @param {Object} evt The event triggered this method.
186      */
187     setValueFromDOM: function(evt) {
188         //this.value = M.Date.create($('#' + this.id + ' ~ div:first').datepicker("getDate"));
189         this.delegateValueUpdate();
190 
191         if((evt === 'change' && this.triggerActionOnChange || evt === 'keyup' && this.triggerActionOnKeyUp) && this.target && this.action) {
192             this.target[this.action](this.value);
193         }
194     }
195 
196 });

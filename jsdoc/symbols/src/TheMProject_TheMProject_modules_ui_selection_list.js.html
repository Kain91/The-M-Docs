 12  * A constant value for single selection mode.
 13  *
 14  * @type String
 15  */
 16 M.SINGLE_SELECTION = 'radio';

 18 /**
 19  * A constant value for multiple selection mode.
 20  *
 21  * @type String
 22  */
 23 M.MULTIPLE_SELECTION = 'checkbox';

 25 /**
 26  * A constant value for single selection mode in a dialog / popup.

 28  * @type String

 34 /**
 35  * @class
 36  *
 37  * This defines the prototype of any selection list view. A selection list view displays
 38  * a list with several items of which either only one single item (M.SINGLE_SELECTION /
 39  * M.SINGLE_SELECTION_DIALOG) or many items (M.MULTIPLE_SELECTION) can be selected.
 40  *
 41  * @extends M.View
 42  */
 43 M.SelectionListView = M.View.extend(
 44 /** @scope M.SelectionListView.prototype */ {
 45 
 46     /**
 47      * The type of this object.
 48      *
 49      * @type String
 50      */
 51     type: 'M.SelectionListView',
 52 
 53     /**
 54      * Determines whether to remove all item if the list is updated or not.
 55      *
 56      * @type Boolean
 57      */
 58     removeItemsOnUpdate: YES,
 59 
 60     /**
 61      * The selection mode for this selection list. This can either be single or
 62      * multiple selection. To set this value use one of the three constants:
 63      *
 64      * - M.SINGLE_SELECTION
 65      *
 66      *   This selection mode will render a selection list with several list items
 67      *   of which only one can be selected. Whenever a new item is selected, the
 68      *   previously selected item automatically gets de-selected. This selection
 69      *   mode's behaviour is equivalent to the plain HTML's radio button.
 70      *
 71      *
 72      * - M.SINGLE_SELECTION_DIALOG
 73      *
 74      *   This selection mode will render a selection list equivalent to the plain
 75      *   HTML's select menu. Only the currently selected item will be visible, and
 76      *   by clicking on this item, the selection list will be displayed in a dialog
 77      *   respectively a popup view. By selecting on of the items, this popup will
 78      *   automatically close and the selected value will be displayed.
 79      *
 80      *
 81      * - M.MULTIPLE_SELECTION
 82      *
 83      *   This selection mode will render a selection list with several list items
 84      *   of which all be selected. So the selection of a new item doesn't lead to
 85      *   automatic de-selected of previously selected items. This selection mode's
 86      *   behaviour is equivalent to the plain HTML's checkboxes.
 87      *
 88      * @type String
 89      */
 90     selectionMode: M.SINGLE_SELECTION,
 91 
 92     /**
 93      * The selected item(s) of this list.
 94      *
 95      * @type String, Array
 96      */
 97     selection: null,
 98 
 99     /**
100      * This property defines the tab bar's name. This is used internally to identify
101      * the selection list inside the DOM.
102      *
103      * @type String
104      */
105     name: null,
106 
107     
108     /**
109      * This property is used to specify an initial value for the selection list if
110      * it is running in 'single selection dialog' (M.SINGLE_SELECTION_DIALOG) mode.
111      * This value is then displayed at startup. You would typically use this e.g. to
112      * specify something like: 'Please select...'.
113      *
114      * As long as this initial value is 'selected', the getSelection() of this selection
115      * list will return nothing. Once a 'real' option is selected, this value is
116      * removed from the selection list.
117      *
118      * Note: This property currently doesn't support non-themed selection lists (see the
119      * applyTheme property).
120      *
121      * @type String
122      */
123     initialText: null,
124 
125     /**
126      * The label proeprty defines a text that is shown above or next to the selection list as a 'title'
127      * for the selection list. e.g. "Name:". If no label is specified, no label will be displayed.
128      *
129      * @type String
130      */
131     label: null,
132 
133     /**
134      * Determines whether to display the selection list grouped with the label specified with the label property.
135      * If set to YES, the selection list and its label are wrapped in a container and styled as a unit 'out of
136      * the box'. If set to NO, custom styling could be necessary.
137      *
138      * @type Boolean
139      */
140     isGrouped: NO,
141 
142     /**
143      * This property specifies the recommended events for this type of view.
144      *
145      * @type Array
146      */
147     recommendedEvents: ['change'],

149     /**
150      * Renders a selection list.
151      *
152      * @private
153      * @returns {String} The selection list view's html representation.
154      */
155     render: function() {
156 
157         this.html += '<div id="' + this.id + '_container"';
158 
159         if(this.isGrouped) {
160             this.html += ' data-role="fieldcontain"';
161         }
162 
163         if(this.cssClass) {
164             this.html += ' class="';
165             var cssClasses = $.trim(this.cssClass).split(' ');            
166             for(var i in cssClasses) {
167                 this.html += (i > 0 ? ' ' : '') + cssClasses[i] + '_container';
168             }
169             this.html += '"';
170         }
171 
172         this.html += '>';
173 
174         if(this.selectionMode === M.SINGLE_SELECTION_DIALOG) {
175             
176             if(this.label) {
177                 this.html += '<label for="' + this.id + '">' + this.label + '</label>';
178             }
179 
180             this.html += '<select name="' + (this.name ? this.name : this.id) + '" id="' + this.id + '"' + this.style() + '>';
181             //this.html += '<select data-native-menu="' + !this.applyTheme + '" name="' + (this.name ? this.name : this.id) + '" id="' + this.id + '"' + this.style() + '>';
182 
183             this.renderChildViews();
184 
185             this.html += '</select>';
186 
187         } else {
188 
189             this.html += '<fieldset data-role="controlgroup" data-native-menu="' + !this.applyTheme + '" id="' + this.id + '">';
190 
191             if(this.label) {
192                 this.html += '<legend>' + this.label + '</legend>';
193             }
194 
195             this.renderChildViews();
196 
197             this.html += '</fieldset>';
198 
199         }
200 
201         this.html += '</div>';
202 
203         return this.html;
204     },

206     /**
207      * Triggers render() on all children of type M.ButtonView based on the specified
208      * selection mode (single or multiple selection).
209      *
210      * @private
211      */
212     renderChildViews: function() {
213         if(this.childViews) {
214             var childViews = $.trim(this.childViews).split(' ');
215 
216             for(var i in childViews) {
217                 var view = this[childViews[i]];
218                 if(view.type === 'M.SelectionListItemView') {
219                     view.parentView = this;
220                     view._name = childViews[i];
221                     this.html += view.render();
222                 } else {
223                     M.Logger.log('Invalid child views specified for SelectionListView. Only SelectionListItemViews accepted.', M.WARN);
224                 }
225             }
226         } else if(!this.contentBinding) {
227             M.Logger.log('No SelectionListItemViews specified.', M.WARN);
228         }
229     },
230 
231     /**
232      * This method is responsible for registering events for view elements and its child views. It
233      * basically passes the view's event-property to M.EventDispatcher to bind the appropriate
234      * events.
235      *
236      * It extend M.View's registerEvents method with some special stuff for text field views and
237      * their internal events.
238      */
239     registerEvents: function() {
240         this.internalEvents = {
241             change: {
242                 target: this,
243                 action: 'itemSelected'
244             }
245         }
246         this.bindToCaller(this, M.View.registerEvents)();
247     },

250      * This method adds a new selection list item to the selection list view by simply appending
251      * its html representation to the selection list view inside the DOM. This method is based
252      * on jQuery's append().
253      *
254      * @param {String} item The html representation of a selection list item to be added.
255      */
256     addItem: function(item) {
257         $('#' + this.id).append(item);
258     },

260     /**
261      * This method removes all of the selection list view's items by removing all of its content in
262      * the DOM. This method is based on jQuery's empty().
263      */
264     removeAllItems: function() {
265         $('#' + this.id).empty();
266     },

268     /**
269      * Updates the the selection list view by re-rendering all of its child views, respectively its
270      * item views.
271      *
272      * @private
273      */
274     renderUpdate: function() {
275         if(this.removeItemsOnUpdate || this.selectionMode === M.SINGLE_SELECTION_DIALOG) {
276             this.removeAllItems();
277 
278             if(this.label && !(this.selectionMode === M.SINGLE_SELECTION_DIALOG)) {
279                 this.addItem('<legend>' + this.label + '</legend>');
280             } else if(this.selectionMode === M.SINGLE_SELECTION_DIALOG) {
281                 if(this.label) {
282                     //this.addItem('<label for="' + this.id + '">' + this.label + '</label>');
283                 }
284                 //this.addItem('<select id="' + this.id + '" onchange="M.EventDispatcher.onClickEventDidHappen(\'click\', \'' + this.id + '\');"></select>');
285             }
286         }
287         
288         if(this.contentBinding) {
289             var items = eval(this.contentBinding);
290             for(var i in items) {
291                 var item  = items[i];
292                 var obj = null;
293                 if(this.selectionMode === M.SINGLE_SELECTION_DIALOG) {
294                     obj = M.SelectionListItemView.design({
295                         value: item.value ? item.value : item,
296                         label: item.label ? item.label : (item.value ? item.value : item),
297                         parentView: this,
298                         isSelected: item.isSelected
299                     });
300                 } else {
301                     obj = M.SelectionListItemView.design({
302                         value: item.value,
303                         label: item.label,
304                         name: item.name,
305                         isSelected: item.isSelected,
306                         parentView: this
307                     });
308                 }
309 
310                 this.addItem(obj.render());
311             }
312             this.themeUpdate();
313         }
314     },

316     /**
317      * Triggers the rendering engine, jQuery mobile, to style the selection list.
318      *
319      * @private
320      */
321     theme: function() {
322         if(this.selectionMode === M.SINGLE_SELECTION_DIALOG && this.applyTheme) {
323             $('#' + this.id).selectmenu();
324             if(this.initialText) {
325                 var parent = $('#' + this.id).parent()
326                 if(parent) {
327                     var label = parent.find('span.ui-btn-text');
328                     if(label) {
329                         label.html(this.initialText);
330                     }
331                 }
332             }
333         } else if(this.selectionMode !== M.SINGLE_SELECTION_DIALOG && this.applyTheme) {
334             $('#' + this.id).controlgroup();
335         }
336     },
337 
338     /**
339      * Triggers the rendering engine, jQuery mobile, to style the selection list.
340      *
341      * @private
342      */
343     themeUpdate: function() {
344         if(this.selectionMode === M.SINGLE_SELECTION_DIALOG && this.applyTheme) {
345             $('#' + this.id).selectmenu('refresh');
346             if(this.initialText) {
347                 $('#' + this.id + '-button').find('span.ui-btn-text').html(this.initialText);
348             }
349         } else if(this.selectionMode !== M.SINGLE_SELECTION_DIALOG) {
350             $('#' + this.id).controlgroup();
351         }
352     },

355      * Method to append css styles inline to the rendered selection list.

357      * @private

361         var html = '';
362         if(this.cssClass) {
363             html += ' class="' + this.cssClass + '"';
364         }
365         return html;

368     /**
369      * This method is called everytime a item is selected / clicked. If the selected item
370      * changed, the defined onSelect action is triggered.
371      *
372      * @param {String} id The id of the selected item.
373      */
374     itemSelected: function(id, event, nextEvent) {
375         var item = null;
376         
377         if(this.selectionMode === M.SINGLE_SELECTION) {
378             item = M.ViewManager.getViewById($('input[name=' + (this.name ? this.name : this.id) + ']:checked').attr('id'));
379             
380             if(item !== this.selection) {
381                 this.selection = item;
382 
383                 if(nextEvent) {
384                     M.EventDispatcher.callHandler(nextEvent, event, NO, [this.selection.value, this.selection]);
385                 }
386             }
387         } else if(this.selectionMode === M.SINGLE_SELECTION_DIALOG) {
388             item = M.ViewManager.getViewById($('#' + this.id + ' :selected').attr('id'));
389 
390             if(item !== this.selection) {
391                 this.selection = item;
392 
393                 $('#' + this.id + '_container').find('.ui-btn-text').html(item.label ? item.label : item.value);
394 
395                 if(nextEvent) {
396                     M.EventDispatcher.callHandler(nextEvent, event, NO, [this.selection.value, this.selection]);
397                 }
398             }
399         } else if(this.selectionMode === M.MULTIPLE_SELECTION) {
400             /*$('#' + id).find('input:checked').each(function() {
401                 $(this).attr('checked', YES).checkboxradio('refresh');
402             });*/
403             /*this.selection = [];
404 
405             var that = this;
406             _.each($('input[name=' + (that.name ? that.name : that.id) + ']:checked'), function(item) {
407                 that.selection.push(M.ViewManager.getViewById(item.id));
408             });*/
409         }
410     },

412     /**
413      * This method returns the selected item's value(s) either as a String (M.SINGLE_SELECTION)
414      * or as an Array (M.MULTIPLE_SELECTION).
415      *
416      * @param {Boolean} returnObject Determines whether to return the selected item(s) as object or not.
417      * @returns {String, Object, Array} The selected item's value(s).
418      */
419     getSelection: function(returnObject) {
420         if(this.selectionMode === M.SINGLE_SELECTION || this.selectionMode === M.SINGLE_SELECTION_DIALOG) {
421             if(this.selection) {
422                 if(returnObject) {
423                     return this.selection;
424                 } else {
425                     return this.selection.value;
426                 }
427             }
428         } else {
429             if(this.selection) {
430                 var selection = [];
431                 _.each(this.selection, function(item) {
432                     if(returnObject) {
433                         selection.push(item);
434                     } else {
435                         selection.push(item.value);
436                     }
437                 });
438                 return selection;
439             }
440         }
441     },

443     /**
444      * This method can be used to select items programmatically. The given parameter can either
445      * be a String (M.SINGLE_SELECTION) or an Array (M.MULTIPLE_SELECTION).
446      *
447      * @param {String, Array} selection The selection that should be applied to the selection list.
448      */
449     setSelection: function(selection) {
450         var that = this;
451         if(this.selectionMode === M.SINGLE_SELECTION && typeof(selection) === 'string') {
452             $('#' + this.id).find('input').each(function() {
453                 var item = M.ViewManager.getViewById($(this).attr('id'));
454                 if(item.value === selection) {
455                     that.removeSelection();
456                     item.isSelected = YES;
457                     that.selection = item;
458                     $(this).attr('checked', 'checked');
459                     if(that.applyTheme) {
460                         $(this).siblings('label:first').addClass('ui-btn-active');
461                         $(this).siblings('label:first').find('span .ui-icon-radio-off').addClass('ui-icon-radio-on');
462                         $(this).siblings('label:first').find('span .ui-icon-radio-off').removeClass('ui-icon-radio-off');
463                     }
464                 }
465             });
466         } else if(this.selectionMode === M.SINGLE_SELECTION_DIALOG && typeof(selection) === 'string') {
467             if(this.applyTheme) {
468                 $('#' + this.id).find('option').each(function() {
469                     var item = M.ViewManager.getViewById($(this).attr('id'));
470                     if(item.value === selection) {
471                         that.removeSelection();
472                         item.isSelected = YES;
473                         that.selection = item;
474                         $('#' + that.id).val(item.value);
475                         if(that.initialText && $('#' + that.id + '-button').find('span.ui-btn-text').html() === that.initialText) {
476                             $('#' + that.id + '-button').find('span.ui-btn-text').html(item.label ? item.label : item.value);
477                         }
478                     }
479                 });
480                 this.initialText = null;
481                 $('#' + this.id).selectmenu('refresh');
482             } else {
483                 $('#' + this.id).find('option').each(function() {
484                     var item = M.ViewManager.getViewById($(this).attr('id'));
485                     if(item.value === selection) {
486                         item.isSelected = YES;
487                         that.selection = item;
488                         $(this).attr('selected', 'selected');
489                     } else {
490                         item.isSelected = NO;
491                         $(this).attr('selected', '');
492                     }
493                 });
494             }
495         } else if(typeof(selection) === 'object') {
496             var removedItems = NO;
497             $('#' + this.id).find('input').each(function() {
498                 var item = M.ViewManager.getViewById($(this).attr('id'));
499                 for(var i in selection) {
500                     var selectionItem = selection[i];
501                     if(item.value === selectionItem) {
502                         if(!removedItems) {
503                             that.removeSelection();
504                             removedItems = YES;
505                         }
506                         item.isSelected = YES;
507                         that.selection.push(item);
508                         $(this).attr('checked', 'checked');
509                         if(that.applyTheme) {
510                             $(this).siblings('label:first').addClass('ui-btn-active');
511                             $(this).siblings('label:first').find('span .ui-icon-checkbox-off').addClass('ui-icon-checkbox-on');
512                             $(this).siblings('label:first').find('span .ui-icon-checkbox-off').removeClass('ui-icon-checkbox-off');
513                         }
514                     }
515                 }
516             });
517         }
518         that.theme();
519     },

521     /**
522      * This method de-selects all of the selection list's items.
523      */
524     removeSelection: function() {
525         var that = this;
526         var type = null;
527         if(this.selectionMode === M.SINGLE_SELECTION || this.selectionMode === M.SINGLE_SELECTION_DIALOG) {
528             this.selection = null;
529 
530             if(this.selectionMode === M.SINGLE_SELECTION) {
531                 type = 'radio';
532             } else {
533                 type = 'select';
534             }
535         } else {
536             this.selection = [];
537             type = 'checkbox';
538         }
539         
540         if(type !== 'select') {
541             $('#' + this.id).find('input').each(function() {
542                 var item = M.ViewManager.getViewById($(this).attr('id'));
543                 item.isSelected = NO;
544                 $(this).removeAttr('checked');
545                 if(that.applyTheme) {
546                     $(this).siblings('label:first').removeClass('ui-btn-active');
547                     $(this).siblings('label:first').find('span .ui-icon-' + type + '-on').addClass('ui-icon-' + type + '-off');
548                     $(this).siblings('label:first').find('span .ui-icon-' + type + '-on').removeClass('ui-icon-' + type + '-on');
549                 }
550             });
551         } else {
552             $('#' + this.id).find('option').each(function() {
553                 var item = M.ViewManager.getViewById($(this).attr('id'));
554                 item.isSelected = NO;
555             });
556         }
557     }

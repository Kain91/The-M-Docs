 11 /**
 12  * @class
 13  *
 14  * M.SearchBarView defines a prototype of a search bar that can be used inside of a list
 15  * view or independently as a plain input field with a search styling.
 16  *
 17  * @extends M.View
 18  */
 19 M.SearchBarView = M.View.extend(
 20 /** @scope M.SearchBarView.prototype */ {
 21 
 22     /**
 23      * The type of this object.
 24      *
 25      * @type String
 26      */
 27     type: 'M.SearchBarView',
 28 
 29     /**
 30      * Determines whether the search bar is part of a list view.
 31      *
 32      * @type Boolean
 33      */
 34     isListViewSearchBar: NO,
 35 
 36     /**
 37      * If the search bar belongs to a list view, this property contains this
 38      * list view.
 39      *
 40      * @type M.ListView
 41      */
 42     listView: null,
 43 
 44     /**
 45      * The initial text shown inside the search bar field describing the input or making a suggestion for
 46      * input e.g. "Please enter your Name."
 47      *
 48      * @type String
 49      */
 50     initialText: '',
 51 
 52     /**
 53      * This property specifies the recommended events for this type of view.
 54      *
 55      * @type Array
 56      */
 57     recommendedEvents: ['focus', 'blur', 'enter', 'keyup'],
 58 
 59     /**
 60      * Renders a search bar.
 61      *
 62      * @private
 63      * @returns {String} The search bar view's html representation.
 64      */
 65     render: function() {
 66         this.html += '<form role="search"' + this.style() + '>';
 67 
 68         this.html += '<input id="' + this.id + '" data-type="search" value="' + (this.value ? this.value : this.initialText) + '" class="' + this.cssClass + '" />';
 69 
 70         this.html += '</form>';
 71 
 72         return this.html;
 73     },
 74 
 75     /**
 76      * This method is responsible for registering events for view elements and its child views. It
 77      * basically passes the view's event-property to M.EventDispatcher to bind the appropriate
 78      * events.
 79      *
 80      * It extend M.View's registerEvents method with some special stuff for text field views and
 81      * their internal events.
 82      */
 83     registerEvents: function() {
 84         this.internalEvents = {
 85             focus: {
 86                 target: this,
 87                 action: 'gotFocus'
 88             },
 89             blur: {
 90                 target: this,
 91                 action: 'lostFocus'
 92             },
 93             keyup: {
 94                 target: this,
 95                 action: 'setValueFromDOM'
 96             }
 97         }
 98         this.bindToCaller(this, M.View.registerEvents)();
 99     },
100 
101     /**
102      * Updates a SearchBarView with DOM access by jQuery.
103      *
104      * @private
105      */
106     renderUpdate: function() {
107         $('#' + this.id).val(this.value);
108         this.styleUpdate();
109     },
110 
111     /**
112      * This method sets its value to the value it has in its DOM representation
113      * and then delegates these changes to a controller property if the
114      * contentBindingReverse property is set.
115      *
116      * Additionally call target / action if set.
117      *
118      * @param {Object} evt The event triggered this method.
119      */
120     setValueFromDOM: function(id, event, nextEvent) {
121         this.value = this.secure($('#' + this.id).val());
122         this.delegateValueUpdate();
123 
124         if(nextEvent) {
125             M.EventDispatcher.callHandler(nextEvent, event, YES);
126         }
127     },
128 
129     /**
130      * Applies some style-attributes to the button.
131      *
132      * @private
133      * @returns {String} The search bar's styling as html representation.
134      */
135     style: function() {
136         var html = '';
137         if(this.isListViewSearchBar) {
138             html += ' class="ui-listview-filter"';
139         }
140         return html;
141     },
142 
143     /**
144      * Method to append css styles inline to the rendered view on the fly.
145      *
146      * @private
147      */
148     styleUpdate: function() {
149         if(this.isInline) {
150             $('#' + this.id).attr('display', 'inline');
151         } else {
152             $('#' + this.id).removeAttr('display');
153         }
154 
155         if(!this.isEnabled) {
156             $('#' + this.id).attr('disabled', 'disabled');
157         } else {
158             $('#' + this.id).removeAttr('disabled');
159         }
160     },
161 
162     /**
163      * This method is called whenever the view gets the focus.
164      * If there is a initial text specified and the value of this search bar field
165      * still equals this initial text, the value is emptied.
166      */
167     gotFocus: function() {
168         if(this.initialText && (!this.value || this.initialText === this.value)) {
169             this.setValue('');
170             if(this.cssClassOnInit) {
171                 this.removeCssClass(this.cssClassOnInit);
172             }
173         }
174         this.hasFocus = YES;
175     },
176 
177     /**
178      * This method is called whenever the view lost the focus.
179      * If there is a initial text specified and the value of this search bar field
180      * is empty, the value is set to the initial text.
181      */
182     lostFocus: function() {
183         if(this.initialText && !this.value) {
184             this.setValue(this.initialText, NO);
185             this.value = '';
186             if(this.cssClassOnInit) {
187                 this.addCssClass(this.cssClassOnInit);
188             }
189         }
190         this.hasFocus = NO;
191     },
192 
193     /**
194      * This method sets the text field's value, initiates its re-rendering
195      * and call the delegateValueUpdate().
196      *
197      * @param {String} value The value to be applied to the text field view.
198      * @param {Boolean} delegateUpdate Determines whether to delegate this value update to any observer or not.
199      */
200     setValue: function(value, delegateUpdate) {
201         this.value = value;
202         this.renderUpdate();
203 
204         if(delegateUpdate) {
205             this.delegateValueUpdate();
206         }
207     },
208 
209     /**
210      * Triggers the rendering engine, jQuery mobile, to style the search bar field.
211      *
212      * @private
213      */
214     theme: function() {
215         if(this.initialText && !this.value && this.cssClassOnInit) {
216             this.addCssClass(this.cssClassOnInit);
217         }
218     }
219 
220 });

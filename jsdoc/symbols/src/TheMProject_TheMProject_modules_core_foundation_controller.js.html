 11 /* Available transitions for page changes */
 12 M.TRANSITION = {};
 13 M.TRANSITION.NONE = 'none';
 14 M.TRANSITION.SLIDE = 'slide';
 15 M.TRANSITION.SLIDEUP = 'slideup';
 16 M.TRANSITION.SLIDEDOWN = 'slidedown';
 17 M.TRANSITION.POP = 'pop';
 18 M.TRANSITION.FADE = 'fade';
 19 M.TRANSITION.FLIP = 'flip';
 20 
 21 m_require('core/foundation/observable.js');
 22 
 23 /**
 24  * @class
 25  *
 26  * The root class for every controller.
 27  *
 28  * Controllers, respectively their properties, are observables. Views can observe them.
 29  *
 30  * @extends M.Object
 31  */
 32 M.Controller = M.Object.extend(
 33 /** @scope M.Controller.prototype */ {
 34 
 35     /**
 36      * The type of this object.
 37      *
 38      * @type String
 39      */
 40     type: 'M.Controller',
 41 
 42     /**
 43      * Makes the controller's properties observable.
 44      */
 45     observable: null,
 46 
 47     /**
 48      * Helper function to build the location href for the view to be displayed.
 49      *
 50      * @param {String} id The id of the new target.
 51      */
 52     buildLocationHref: function(id) {
 53         return location.pathname + '#' + id;
 54     },
 55 
 56     /**
 57      * Switch the active tab in the application. This includes both activating this tab
 58      * visually and switching the page.
 59      *
 60      * @param {M.TabBarView} tab The tab to be activated.
 61      */
 62     switchToTab: function(tab) {
 63         var currentTab = tab.parentView.activeTab;
 64         var newPage = M.ViewManager.getPage(tab.page);
 65 
 66         /* store the active tab in tab bar view */
 67         tab.parentView.setActiveTab(tab);
 68 
 69         if(tab === currentTab) {
 70             var currentPage = M.ViewManager.getCurrentPage();
 71             if(currentPage !== newPage) {
 72                 this.switchToPage(newPage, null, YES, NO);
 73             }
 74         } else {
 75             this.switchToPage(newPage, M.TRANSITION.NONE, NO, YES);
 76         }
 77     },
 78 
 79     /**
 80      * Switch the active page in the application.
 81      *
 82      * @param {Object, String} page The page to be displayed or its name.
 83      * @param {String} transition The transition that should be used. Default: horizontal slide
 84      * @param {Boolean} isBack YES will cause a reverse-direction transition. Default: NO
 85      * @param {Boolean} changeLoc Update the browser history. Default: YES
 86      */
 87     switchToPage: function(page, transition, isBack, changeLoc) {
 88         var timeStart = M.Date.now();
 89         page = page && typeof(page) === 'object' ? page : M.Application.viewManager.getPage(page);
 90 
 91         if(page) {
 92             transition = transition ? transition : M.TRANSITION.SLIDE;
 93             isBack = isBack !== undefined ? isBack : NO;
 94             changeLoc = changeLoc !== undefined ? changeLoc : YES;
 95 
 96             /* Now do the page change by using a jquery mobile method and pass the properties */
 97             if(page.type === 'M.PageView') {
 98                 //console.log('$.mobile.changePage(' + page.id + ', ' + (M.Application.useTransitions ? transition : M.TRANSITION.NONE) + ', ' + (M.Application.useTransitions ? isBack : NO) + ', ' + changeLoc + ');');
 99                 $.mobile.changePage(page.id, {
100                     transition: M.Application.useTransitions ? transition : M.TRANSITION.NONE,
101                     reverse: M.Application.useTransitions ? isBack : NO,
102                     changeHash: YES
103                 });
104             }
105 
106             /* Save the current page in the view manager */
107             M.Application.viewManager.setCurrentPage(page);
108         } else {
109             M.Logger.log('Page "' + page + '" not found', M.ERR);
110         }
111     },
112 
113     /**
114      * Returns the class property behind the given key and informs its observers.
115      *
116      * @param {String} key The key of the property to be changed.
117      * @param {Object, String} value The value to be set.
118      */
119     set: function(key, value) {
120         this[key] = value;
121         if(!this.observable) {
122             return;
123         }
124         this.observable.notifyObservers(key);
125     }
126 
127 });

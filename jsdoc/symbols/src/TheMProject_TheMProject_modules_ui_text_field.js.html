 53 /**
 54  * @class
 55  *
 56  * M.TextFieldView is the prototype of any text field input view. It can be rendered as both
 57  * a single line text field and a multiple line text field. If it is styled as a multiple
 58  * line text field, is has a built-in autogrow mechanism so the textfield is getting larger
 59  * depending on the number of lines of text a user enters.
 60  *
 61  * @extends M.View
 62  */
 63 M.TextFieldView = M.View.extend(
 64 /** @scope M.TextFieldView.prototype */ {
 65 
 66    /**
 67      * The type of this object.
 68      *
 69      * @type String
 70      */
 71     type: 'M.TextFieldView',
 72 
 73    /**
 74     * The name of the text field. During the rendering, this property gets assigned to the name
 75     * property of the text field's html representation. This can be used to manually access the
 76     * text field's DOM representation later on.
 77     *
 78     * @type String
 79     */
 80     name: null,
 81 
 82     /**
 83      * The label proeprty defines a text that is shown above or next to the textfield as a 'title'
 84      * for the textfield. e.g. "Name:". If no label is specified, no label will be displayed.
 85      *
 86      * @type String
 87      */
 88     label: null,
 89 
 90     /**
 91      * The initial text shown inside the text field describing the input or making a suggestion for input
 92      * e.g. "Please enter your Name."
 93      *
 94      * @type String
 95      */
 96     initialText: '',
 97 
 98     /**
 99      * Determines whether to display the textfield grouped with the label specified with the label property.
100      * If set to YES, the textfield and its label are wrapped in a container and styled as a unit 'out of
101      * the box'. If set to NO, custom styling could be necessary.
102      *
103      * If there is no label specified, this property is ignored by default.
104      *
105      * @type Boolean
106      */
107     isGrouped: NO,
108 
109     /**
110      * Defines whether the text field has multiple lines respectively is a text area.
111      *
112      * @type Boolean
113      */
114     hasMultipleLines: NO,
115 
116     /**
117      * A boolean value to determine whether a view has a value or not. A text field view does have
118      * a value, so it is set to YES.
119      *
120      * @type Boolean
121      */
122     hasValue: YES,
123 
124     /**
125      * This property specifies the input type of this input field. Possible values are:
126      *
127      *   - M.INPUT_TEXT --> text input (default)
128      *   - M.INPUT_PASSWORD --> password
129      *   - M.INPUT_NUMBER --> number
130      *   - M.INPUT_TELEPHONE --> tel
131      *   - M.INPUT_URL --> url
132      *   - M.INPUT_EMAIL --> email
133      *
134      * Note, that these types are not yet supported by all browsers!
135      *
136      * @type String
137      */
138     inputType: M.INPUT_TEXT,
139 
140     /**
141      * This property specifies the recommended events for this type of view.
142      *
143      * @type Array
144      */
145     recommendedEvents: ['focus', 'blur', 'enter', 'keyup'],
146 
147     /**
148      * Renders a TextFieldView
149      * 
150      * @private
151      * @returns {String} The text field view's html representation.
152      */
153     render: function() {
154         this.html += '<div';
155 
156         if(this.label && this.isGrouped) {
157             this.html += ' data-role="fieldcontain"';
158         }
159 
160         if(this.cssClass) {
161             this.html += ' class="' + this.cssClass + '_container"';
162         }
163 
164         this.html += '>';
165 
166         if(this.label) {
167             this.html += '<label for="' + (this.name ? this.name : this.id) + '">' + this.label + '</label>';
168         }
169 
170         if(this.hasMultipleLines) {
171             this.html += '<textarea cols="40" rows="8" name="' + (this.name ? this.name : this.id) + '" id="' + this.id + '"' + this.style() + '>' + (this.value ? this.value : this.initialText) + '</textarea>';
172             
173         } else {
174             this.html += '<input type="' + this.inputType + '" name="' + (this.name ? this.name : this.id) + '" id="' + this.id + '"' + this.style() + ' value="' + (this.value ? this.value : this.initialText) + '" />';
175         }
176 
177         this.html += '</div>';
178 
179         return this.html;
180     },
181 
182     /**
183      * This method is responsible for registering events for view elements and its child views. It
184      * basically passes the view's event-property to M.EventDispatcher to bind the appropriate
185      * events.
186      *
187      * It extend M.View's registerEvents method with some special stuff for text field views and
188      * their internal events.
189      */
190     registerEvents: function() {
191         this.internalEvents = {
192             focus: {
193                 target: this,
194                 action: 'gotFocus'
195             },
196             blur: {
197                 target: this,
198                 action: 'lostFocus'
199             },
200             keyup: {
201                 target: this,
202                 action: 'setValueFromDOM'
203             }
204         }
205         this.bindToCaller(this, M.View.registerEvents)();
206     },
207 
208     /**
209      * The contentDidChange method is automatically called by the observable when the
210      * observable's state did change. It then updates the view's value property based
211      * on the specified content binding.
212      *
213      * This is a special implementation for M.TextFieldView.
214      */
215     contentDidChange: function(){
216         var contentBinding = this.contentBinding ? this.contentBinding : (this.computedValue) ? this.computedValue.contentBinding : null;
217 
218         if(!contentBinding || (this.hasFocus)) {
219             return;
220         }
221 
222         if(this.contentBinding) {
223             this.value = contentBinding.target[contentBinding.property];
224         } else if(this.computedValue.contentBinding) {
225             this.computedValue.value = contentBinding.target[contentBinding.property];
226         }
227 
228         this.renderUpdate();
229         this.delegateValueUpdate();
230 
231         /* TODO: ADD CONTENT BINDING FOR MORE THAN ONE LEVEL */
232     },
233 
234     /**
235      * Updates a TextFieldView with DOM access by jQuery.
236      *
237      * @private
238      */
239     renderUpdate: function() {
240         $('#' + this.id).val(this.value);
241         this.styleUpdate();
242     },
243 
244     /**
245      * This method is called whenever the view gets the focus.
246      * If there is a initial text specified and the value of this text field
247      * still equals this initial text, the value is emptied.
248      *
249      * @param {String} id The DOM id of the event target.
250      * @param {Object} event The DOM event.
251      * @param {Object} nextEvent The next event (external event), if specified.
252      */
253     gotFocus: function(id, event, nextEvent) {
254         if(this.initialText && (!this.value || this.initialText === this.value)) {
255             this.setValue('');
256             if(this.cssClassOnInit) {
257                 this.removeCssClass(this.cssClassOnInit);
258             }
259         }
260         this.hasFocus = YES;
261 
262         if(nextEvent) {
263             M.EventDispatcher.callHandler(nextEvent, event, YES);
264         }
265     },
266 
267     /**
268      * This method is called whenever the view lost the focus.
269      * If there is a initial text specified and the value of this text field
270      * is empty, the value is set to the initial text.
271      *
272      * @param {String} id The DOM id of the event target.
273      * @param {Object} event The DOM event.
274      * @param {Object} nextEvent The next event (external event), if specified.
275      */
276     lostFocus: function(id, event, nextEvent) {
277         if(this.initialText && !this.value) {
278             this.setValue(this.initialText, NO);
279             this.value = '';
280             if(this.cssClassOnInit) {
281                 this.addCssClass(this.cssClassOnInit);
282             }
283         }
284         this.hasFocus = NO;
285 
286         if(nextEvent) {
287             M.EventDispatcher.callHandler(nextEvent, event, YES);
288         }
289     },
290 
291     /**
292      * Method to append css styles inline to the rendered text field.
293      *
294      * @private
295      * @returns {String} The text field's styling as html representation.
296      */
297     style: function() {
298         var html = ' style="';
299         if(this.isInline) {
300             html += 'display:inline;';
301         }
302         if(!this.isEnabled) {
303             html += 'disabled:disabled;';
304         }
305         html += '"';
306         
307         if(this.cssClass) {
308             html += ' class="' + this.cssClass + '"';
309         }
310 
311         return html;
312     },
313 
314     /**
315      * Triggers the rendering engine, jQuery mobile, to style the text field.
316      *
317      * @private
318      */
319     theme: function() {
320         if(this.initialText && !this.value && this.cssClassOnInit) {
321             this.addCssClass(this.cssClassOnInit);
322         }
323 
324         /* trigger keyup event to make the text field autogrow */
325         if(this.value) {
326             $('#'  + this.id).trigger('keyup');
327         }
328     },
329 
330     /**
331      * Method to append css styles inline to the rendered view on the fly.
332      *
333      * @private
334      */
335     styleUpdate: function() {
336         if(this.isInline) {
337             $('#' + this.id).attr('display', 'inline');
338         } else {
339             $('#' + this.id).removeAttr('display');
340         }
341 
342         if(!this.isEnabled) {
343             $('#' + this.id).attr('disabled', 'disabled');
344         } else {
345             $('#' + this.id).removeAttr('disabled');
346         }
347 
348         /* trigger keyup event to make the text field autogrow */
349         if(this.value) {
350             $('#'  + this.id).trigger('keyup');
351         }
352     },
353 
354     /**
355      * This method sets its value to the value it has in its DOM representation
356      * and then delegates these changes to a controller property if the
357      * contentBindingReverse property is set.
358      *
359      * Additionally call target / action if set.
360      *
361      * @param {String} id The DOM id of the event target.
362      * @param {Object} event The DOM event.
363      * @param {Object} nextEvent The next event (external event), if specified.
364      */
365     setValueFromDOM: function(id, event, nextEvent) {
366         this.value = this.secure($('#' + this.id).val());
367         this.delegateValueUpdate();
368 
369         if(nextEvent) {
370             M.EventDispatcher.callHandler(nextEvent, event, YES);
371         }
372     },
373 
374     /**
375      * This method sets the text field's value, initiates its re-rendering
376      * and call the delegateValueUpdate().
377      *
378      * @param {String} value The value to be applied to the text field view.
379      * @param {Boolean} delegateUpdate Determines whether to delegate this value update to any observer or not.
380      */
381     setValue: function(value, delegateUpdate) {
382         this.value = value;
383         this.renderUpdate();
384 
385         if(delegateUpdate) {
386             this.delegateValueUpdate();
387         }
388     },
389 
390     /**
391      * This method disables the text field by setting the disabled property of its
392      * html representation to true.
393      */
394     disable: function() {
395         this.isEnabled = NO;
396         this.renderUpdate();
397     },
398 
399     /**
400      * This method enables the text field by setting the disabled property of its
401      * html representation to false.
402      */
403     enable: function() {
404         this.isEnabled = YES;
405         this.renderUpdate();
406     },
407 
408     /**
409      * This method clears the text field's value, both in the DOM and within the JS object.
410      */
411     clearValue: function() {
412         this.setValue('');
413 
414         /* call lostFocus() to get the initial text displayed */
415         this.lostFocus();
416     }
417 
418 });

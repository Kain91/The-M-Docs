 11 /**
 12  * @class
 13  *
 14  * This defines the prototype for any button view. A button is a view element that is
 15  * typically used for triggering an action, e.g. switching to another page, firing a
 16  * request or opening a dialog.
 17  *
 18  * @extends M.View
 19  */
 20 M.ButtonView = M.View.extend(
 21 /** @scope M.ButtonView.prototype */ {
 22 
 23     /**
 24      * The type of this object.
 25      *
 26      * @type String
 27      */
 28     type: 'M.ButtonView',
 29 
 30     /**
 31      * Determines whether this button is active or not.
 32      *
 33      * Note: This property is only used if the button is part of a button group (M.ButtonGroupView).
 34      *
 35      * @type Boolean
 36      */
 37     isActive: NO,
 38 
 39     /**
 40      * Determines whether to display the button ony with an icon but no text or not.
 41      *
 42      * @type Boolean
 43      */
 44     isIconOnly: NO,
 45 
 46     /**
 47      * This property can be used to specify a certain hyperlink type for this button. It only
 48      * works in combination with the hyperlinkTarget property.
 49      *
 50      * @type String
 51      */
 52     hyperlinkType: null,
 53 
 54     /**
 55      * This property can be used to specify a hyperlink target for this button. It only
 56      * works in combination with the hyperlinkType property.
 57      *
 58      * @type String
 59      */
 60     hyperlinkTarget: null,
 61 
 62     /**
 63      * This property can be used to specify a tag, that is independent from the button's
 64      * value. This allows you to identify a button, without having to worry about changes
 65      * to its value.
 66      *
 67      * @type String
 68      */
 69     tag: null,
 70 
 71     /**
 72      * This property specifies the recommended events for this type of view.
 73      *
 74      * @type Array
 75      */
 76     recommendedEvents: ['click', 'tap'],
 77 
 78     /**
 79      * Renders a button as an input tag. Input is automatically converted by jQuery mobile.
 80      *
 81      * @private
 82      * @returns {String} The button view's html representation.
 83      */
 84     render: function() {
 85         this.computeValue();
 86         this.html += '<a data-role="button" id="' + this.id + '"' + this.style() + ' ';
 87 
 88         if(this.hyperlinkTarget && this.hyperlinkType) {
 89             switch (this.hyperlinkType) {
 90                 case M.HYPERLINK_EMAIL:
 91                     this.html += 'rel="external" href="mailto:' + this.hyperlinkTarget + '"';
 92                     break;
 93                 case M.HYPERLINK_WEBSITE:
 94                     this.html += 'rel="external" target="_blank" href="' + this.hyperlinkTarget + '"';
 95                     break;
 96                 case M.HYPERLINK_PHONE:
 97                     this.html += 'rel="external" href="tel:' + this.hyperlinkTarget + '"';
 98                     break;
 99             }
100         } else {
101             this.html += 'href="#"';
102         }
103 
104         this.html += '>' + this.value + '</a>';
105 
106         return this.html;
107     },
108 
109     /**
110      * Updates the value of the button with DOM access by jQuery.
111      *
112      * @private
113      */
114     renderUpdate: function() {
115         this.computeValue();
116         if(this.applyTheme) {
117             this.theme();
118             $('#' + this.id).parent().find('.ui-btn-text').text(this.value);
119         } else {
120             $('#' + this.id).find('.ui-btn-text').text(this.value);
121         }
122     },
123 
124     /**
125      * Sets the button's value and calls renderUpdate() to make the value update visible.
126      *
127      * @param {String} value The button's new value.
128      */
129     setValue: function(value) {
130         this.value = value;
131         this.renderUpdate();
132     },
133 
134     /**
135      * Triggers the rendering engine, jQuery mobile, to style the button.
136      *
137      * @private
138      */
139     theme: function() {
140         $('#' + this.id).button();
141     },
142 
143     /**
144      * Applies some style-attributes to the button.
145      *
146      * @private
147      * @returns {String} The button's styling as html representation.
148      */
149     style: function() {
150         var html = '';
151         if(this.isInline) {
152             html += ' data-inline="true"';
153         }
154         if(this.icon) {
155             html += ' data-icon="' + this.icon + '"';
156         }
157         if(this.cssClass) {
158             html += ' data-theme="' + this.cssClass + '"';
159         }
160         if(this.isIconOnly) {
161             html += ' data-iconpos="notext"';
162         }
163         if(this.cssStyle) {
164             html += 'style="' + this.cssStyle + '"';
165         }
166         return html;
167     }
168 
169 });

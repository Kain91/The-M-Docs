 11 m_require('core/utility/logger.js');
 12 
 13 /**
 14  * @class
 15  *
 16  * Object for dispatching all incoming events.
 17  *
 18  * @extends M.Object
 19  */
 20 M.EventDispatcher = M.Object.extend(
 21 /** @scope M.EventDispatcher.prototype */ {
 22 
 23     /**
 24      * The type of this object.
 25      *
 26      * @type String
 27      */
 28     type: 'M.EventDispatcher',
 29 
 30     /**
 31      * Saves the latest on click event to make sure that there are no multiple events
 32      * fired for one click.
 33      *
 34      * @type {Object}
 35      */
 36     lastEvent: {},
 37 
 38     /**
 39      * This method is used to register events and link them to a corresponding action.
 40      * 
 41      * @param {String, Object} eventSource The view's id or a DOM object.
 42      * @param {Object} events The events to be registered for the given view or DOM object.
 43      */
 44     registerEvents: function(eventSource, events, recommendedEvents, sourceType) {
 45         if(!events || typeof(events) !== 'object') {
 46             M.Logger.log('No events passed for \'' + eventSource + '\'!', M.WARN);
 47             return;
 48         }
 49 
 50         eventSource = this.getEventSource(eventSource);
 51         if(!this.checkEventSource(eventSource)) {
 52             return;
 53         }
 54 
 55         _.each(events, function(handler, type) {
 56             M.EventDispatcher.registerEvent(type, eventSource, handler, recommendedEvents, sourceType, YES);
 57         });
 58     },
 59 
 60     /**
 61      * This method is used to register a certain event for a certain view or DOM object
 62      * and link them to a corresponding action.
 63      *
 64      * @param {String} type The type of the event.
 65      * @param {String, Object} eventSource The view's id, the view object or a DOM object.
 66      * @param {Object} handler The handler for the event.
 67      * @param {Object} recommendedEvents The recommended events for this event source.
 68      * @param {Object} sourceType The type of the event source.
 69      * @param {Boolean} isInternalCall The flag to determine whether this is an internal call or not.
 70      */
 71     registerEvent: function(type, eventSource, handler, recommendedEvents, sourceType, isInternalCall) {
 72         if(!isInternalCall) {
 73             if(!handler || typeof(handler) !== 'object') {
 74                 M.Logger.log('No event passed!', M.WARN);
 75                 return;
 76             }
 77 
 78             eventSource = this.getEventSource(eventSource);
 79             if(!this.checkEventSource(eventSource)) {
 80                 return;
 81             }
 82         }
 83 
 84         if(!(recommendedEvents && _.indexOf(recommendedEvents, type) > -1)) {
 85             if(sourceType && typeof(sourceType) === 'string') {
 86                 M.Logger.log('Event type \'' + type + '\' not recommended for ' + sourceType + '!', M.WARN);
 87             } else {
 88                 M.Logger.log('Event type \'' + type + '\' not recommended!', M.WARN);
 89             }
 90         }
 91 
 92         if(!this.checkHandler(handler, type)) {
 93             return;
 94         }
 95 
 96         /* switch enter event to keyup with keycode 13 */
 97         if(type === 'enter') {
 98             eventSource.bind('keyup', function(event) {
 99                 if(event.which === 13) {
100                     $(this).trigger('enter');
101                 }
102             });
103         }
104 
105         var that = this;
106         eventSource.unbind(type).bind(type, function(event) {
107             event.preventDefault();
108             event.stopPropagation();
109 
110             /* fix for jqm problem with two times firing pageshow event for the first page */
111             if(type == 'pageshow' || type == 'pagebeforeshow') {
112                 if(that.lastEvent[type] && that.lastEvent[type].timeBetween(M.Date.now()) <= 100) {
113                     return;
114                 } else {
115                     that.lastEvent[type] = M.Date.now();
116                 }
117             }
118 
119             /* fix for jqm problem with multiple tap events */
120             if(type == 'tap') {
121                 if(that.lastEvent[type] && event.clientX + '/' + event.clientY == that.lastEvent[type] ) {
122                     return;
123                 } else {
124                     that.lastEvent[type] = event.clientX + '/' + event.clientY;
125                 }
126             }
127 
128             /* event logger, uncomment for development mode */
129             //M.Logger.log('Event \'' + event.type + '\' did happen for id \'' + event.currentTarget.id + '\'', M.INFO);
130 
131             if(handler.nextEvent) {
132                 that.bindToCaller(handler.target, handler.action, [event.currentTarget.id ? event.currentTarget.id : event.currentTarget, event, handler.nextEvent])();
133             } else {
134                 that.bindToCaller(handler.target, handler.action, [event.currentTarget.id ? event.currentTarget.id : event.currentTarget, event])();
135             }
136         });
137 
138     },
139 
140     /**
141      * This method is used to explicitly call an event handler. We mainly use this for
142      * combining internal and external events.
143      *
144      * @param {Object} handler The handler for the event.
145      * @param {Object} event The original DOM event.
146      * @param {Boolean} passEvent Determines whether or not to pass the event and its target as the first parameters for the handler call.
147      * @param {Array} parameters The (additional) parameters for the handler call.
148      */
149     callHandler: function(handler, event, passEvent, parameters) {
150         if(!this.checkHandler(handler, (event && event.type ? event.type : 'undefined'))) {
151             return;
152         }
153 
154         if(!passEvent) {
155             this.bindToCaller(handler.target, handler.action, parameters)();
156         } else {
157             this.bindToCaller(handler.target, handler.action, [event.currentTarget.id ? event.currentTarget.id : event.currentTarget, event])();
158         }
159     },
160 
161     /**
162      * This method is used to check the handler. It tests if target and action are
163      * specified correctly.
164      *
165      * @param {Object} handler The handler for the event.
166      * @param {String} type The type of the event.
167      * @return {Boolean} Specifies whether or not the check was successful.
168      */
169     checkHandler: function(handler, type) {
170         if(typeof(handler.action) === 'string') {
171             if(handler.target) {
172                 if(handler.target[handler.action] && typeof(handler.target[handler.action]) === 'function') {
173                     handler.action = handler.target[handler.action];
174                     return YES;
175                 } else {
176                     M.Logger.log('No action \'' + handler.action + '\' found for given target and the event type \'' + type + '\'!', M.WARN);
177                     return NO;
178                 }
179             } else {
180                 M.Logger.log('No valid target passed for action \'' + handler.action + '\' and the event type \'' + type + '\'!', M.WARN);
181                 return NO;
182             }
183         } else if(typeof(handler.action) !== 'function') {
184             M.Logger.log('No valid action passed for the event type \'' + type + '\'!', M.WARN);
185             return NO;
186         }
187 
188         return YES;
189     },
190 
191     /**
192      * This method is used to get the event source as a DOM object.
193      *
194      * @param {Object, String} eventSource The event source.
195      * @return {Object} The event source as dom object.
196      */
197     getEventSource: function(eventSource) {
198         if(typeof(eventSource) === 'string') {
199             eventSource = $('#' + eventSource + ':first');
200         } else {
201             eventSource = $(eventSource);
202         }
203         
204         return eventSource;
205     },
206 
207     /**
208      * This method is used to check the event source. It tests if it is correctly
209      * specified.
210      *
211      * @param {Object} eventSource The event source.
212      * @return {Boolean} Specifies whether or not the check was successful.
213      */
214     checkEventSource: function(eventSource) {
215         if(!eventSource) {
216             M.Logger.log('The event source is invalid!', M.WARN);
217             return NO;
218         }
219         
220         return YES;
221     }
222 
223 });
